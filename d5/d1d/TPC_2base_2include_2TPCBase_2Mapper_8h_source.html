62  int getCRU(const Sector& sec, GlobalPadNumber globalPad) const
63  {
64  const auto row = mMapGlobalPadToPadPos[globalPad].getRow();
65  const auto nCRUPerSector = mMapPadRegionInfo.size();
66  int region = 0;
67  for (size_t i = 1; i < nCRUPerSector; ++i) {
68  if (row < mMapPadRegionInfo[i].getGlobalRowOffset()) {
69  break;
70  }
71  ++region;
72  }
73 
74  return int(sec * nCRUPerSector + region);
75  }

381  {
382  const double cs = std::cos(alpha), sn = std::sin(alpha);

569 inline bool Mapper::isOutOfSector(GlobalPosition3D posEle, const Sector& sector, const float margin) const
570 {

572  int secLeft = int(Sector::getLeft(sector));
573  const float dSectorBoundaryRight = -SinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secRight % SECTORSPERSIDE] * posEle.Y();
574  const float dSectorBoundaryLeft = -SinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.X() + CosinsPerSectorNotShifted[secLeft % SECTORSPERSIDE] * posEle.Y();
575 
576  if ((dSectorBoundaryLeft > 0 && dSectorBoundaryRight < 0) || (dSectorBoundaryLeft < 0 && dSectorBoundaryRight > 0)) {
577  return false;
578  }
579  if (std::abs(dSectorBoundaryLeft) > margin && std::abs(dSectorBoundaryRight) > margin) {
580  return true;
581  }
582  return false;
