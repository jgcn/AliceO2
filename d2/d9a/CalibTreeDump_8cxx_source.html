46 void CalibTreeDump::addDefaultMapping(TTree* tree)
47 {
48  // loop over ROCs
49  //
50 
51  // ===| mapper |==============================================================
52  const auto& mapper = Mapper::instance();
53 
54  // ===| default mapping objects |=============================================
55  uint16_t rocNumber = 0;
56  // positions
57  std::vector<float> gx;
58  std::vector<float> gy;
59  std::vector<float> lx;
60  std::vector<float> ly;
61  // row and pad
62  std::vector<unsigned short> row;
63  std::vector<unsigned short> pad;
64  std::vector<unsigned short> cpad;
65 
66  // ===| add branches with default mappings |==================================
67  tree->Branch("roc", &rocNumber);
68  tree->Branch("gx", &gx);
69  tree->Branch("gy", &gy);
70  tree->Branch("lx", &lx);
71  tree->Branch("ly", &ly);
72 
73  // ===| loop over readout chambers |==========================================
74  for (ROC roc; !roc.looped(); ++roc) {
75  rocNumber = roc;
76 
77  // ===| clear position vectors |============================================
78  gx.clear();
79  gy.clear();
80  lx.clear();
81  ly.clear();
82 
83  // ===| loop over pad rows |================================================
84  const int numberOfRows = mapper.getNumberOfRowsROC(roc);
85  for (int irow = 0; irow < numberOfRows; ++irow) {
86 
87  // ===| loop over pads in row |===========================================
88  const int numberOfPadsInRow = mapper.getNumberOfPadsInRowROC(roc, irow);
89  for (int ipad = 0; ipad < numberOfPadsInRow; ++ipad) {
90  const PadROCPos padROCPos(rocNumber, irow, ipad);
91  const PadPos padPos = mapper.getGlobalPadPos(padROCPos); // pad and row in sector
92  const PadCentre& localPadXY = mapper.getPadCentre(padPos);
93  const LocalPosition2D globalPadXY = mapper.getPadCentre(padROCPos);
94 
95  gx.emplace_back(globalPadXY.X());
96  gy.emplace_back(globalPadXY.Y());
97  lx.emplace_back(localPadXY.X());
98  ly.emplace_back(localPadXY.Y());
99 
100  row.emplace_back(irow);
101  pad.emplace_back(ipad);
102  cpad.emplace_back(ipad - numberOfPadsInRow / 2);
103  }
104  }
105 
106  tree->Fill();
107  }
108 }
