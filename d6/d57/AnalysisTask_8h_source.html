301  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
302  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*)(Grouping, Associated...))
303  {
304  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, &C::process);
305  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, &C::process);
306 
307  if constexpr (sizeof...(Associated) == 0) {
308  // No extra tables: we need to either iterate over the contents of
309  // grouping or pass the whole grouping table, depending on whether Grouping
310  // is a o2::soa::Table or a o2::soa::RowView
311  if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Table>::value) {
312  task.process(groupingTable);
313  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::RowViewBase>::value) {
314  for (auto& groupedElement : groupingTable) {
315  task.process(groupedElement);
316  }
317  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Join>::value) {
318  task.process(groupingTable);
319  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Filtered>::value) {
320  task.process(groupingTable);
321  } else {
322  static_assert(always_static_assert_v<Grouping>,
323  "The first argument of the process method of your task must be either"
324  " a o2::soa::Table or a o2::soa::RowView");
325  }
326  } else if constexpr (sizeof...(Associated) == 1) {
327  // One extra table provided: if the first argument itself is a table,
328  // then we simply pass it over to the process method and let the user do the
329  // double looping (e.g. if they want to do some association between different
330  // physics quantities.
331  //
332  // If the first argument is a single element, we consider that the user
333  // wants to do a loop first on the table associated to the first element,
334  // then to the subgroup of the second table which is associated to the
335  // first one. E.g.:
336  //
337  // MyTask::process(Collision const& collision, Tracks const& tracks)
338  //
339  // Will iterate on all the tracks for the provided collision.
340  if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Table>::value) {
341  static_assert((is_specialization<Associated, o2::soa::Table>::value && ...),
342  "You cannot have a soa::RowView iterator as an argument after the "
343  " first argument of type soa::Table which is found as in the "
344  " prototype of the task process method.");
345  task.process(groupingTable, std::get<0>(associatedTables));
346  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::RowViewBase>::value) {
347  using AssociatedType = std::tuple_element_t<0, std::tuple<Associated...>>;
348  if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::RowViewBase>::value) {
349  auto groupedTable = std::get<0>(associatedTables);
350  size_t currentGrouping = 0;
351  Grouping groupingElement = groupingTable.begin();
352  for (auto& groupedElement : groupedTable) {
353  // FIXME: this only works for collisions for now...
354  auto groupingIndex = groupedElement.collisionId(); // Fine for the moment.
355  // We find the associated collision, assuming they are sorted.
356  while (groupingIndex > currentGrouping) {
357  // This const_cast is done because I do not want people to be
358  // able to move the iterator in the user code.
359  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
360  ++const_cast<std::decay_t<AssociatedType>&>(groupedElement);
361  }
362  task.process(groupingElement, groupedElement);
363  }
364  } else if constexpr (is_base_of_template<o2::soa::Table, std::decay_t<AssociatedType>>::value) {
365  auto allGroupedTable = std::get<0>(associatedTables);
366  using groupingMetadata = typename aod::MetadataTrait<std::decay_t<Grouping>>::metadata;
367  arrow::compute::FunctionContext ctx;
368  std::vector<arrow::compute::Datum> groupsCollection;
369  auto indexColumnName = std::string("fID4") + groupingMetadata::label();
370  auto result = o2::framework::sliceByColumn(&ctx, indexColumnName,
371  allGroupedTable.asArrowTable(), &groupsCollection);
372  if (result.ok() == false) {
373  LOGF(ERROR, "Error while splitting second collection");
374  return;
375  }
376  size_t currentGrouping = 0;
377  auto groupingElement = groupingTable.begin();
378 
379  // FIXME: this assumes every groupingElement has a group associated,
380  // which migh not be the case.
381  if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::Table>::value || is_specialization<std::decay_t<AssociatedType>, o2::soa::Filtered>::value) {
382  for (auto& groupedDatum : groupsCollection) {
383  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(groupedDatum.value);
384  task.process(groupingElement, AssociatedType{groupedElementsTable});
385  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
386  }
387  } else if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::Join>::value || is_specialization<std::decay_t<AssociatedType>, o2::soa::Concat>::value) {
388  std::vector<std::shared_ptr<arrow::Table>> tables;
389  for (auto& groupedDatum : groupsCollection) {
390  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(groupedDatum.value);
391  tables.push_back(groupedElementsTable);
392  }
393  task.process(groupingElement, AssociatedType{tables});
394  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
395  }
396  } else {
397  static_assert(always_static_assert_v<AssociatedType>, "I do not know how to iterate on this");
398  }
399  } else {
400  static_assert(always_static_assert_v<Grouping>,
401  "Only grouping by Collision is supported for now");
402  }
403  } else {
404  static_assert(always_static_assert_v<Grouping, Associated...>,
405  "Unable to find a way to iterate on the provided set of arguments. Probably unimplemented");
406  }
407  }

560 template <typename T, typename... Args>
561 DataProcessorSpec adaptAnalysisTask(std::string name, Args&&... args)
562 {
563  auto task = std::make_shared<T>(std::forward<Args>(args)...);
564 
565  std::vector<OutputSpec> outputs;
566  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
567  std::apply([&outputs](auto&... x) { return (OutputManager<std::decay_t<decltype(x)>>::appendOutput(outputs, x), ...); }, tupledTask);
568  static_assert(has_process<T>::value || has_run<T>::value || has_init<T>::value,
569  "At least one of process(...), T::run(...), init(...) must be defined");
570 
571  std::vector<InputSpec> inputs;
572  if constexpr (has_process<T>::value) {
573  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, inputs);
574  }
575 
576  auto algo = AlgorithmSpec::InitCallback{[task](InitContext& ic) {
577  auto& callbacks = ic.services().get<CallbackService>();
578  auto endofdatacb = [task](EndOfStreamContext& eosContext) {
579  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
580  std::apply([&eosContext](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::postRun(eosContext, x), ...); }, tupledTask);
581  eosContext.services().get<ControlService>().readyToQuit(QuitRequest::Me);
582  };
583  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
584 
585  if constexpr (has_init<T>::value) {
586  task->init(ic);
587  }
588  return [task](ProcessingContext& pc) {
589  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
590  std::apply([&pc](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::prepare(pc, x), ...); }, tupledTask);
591  if constexpr (has_run<T>::value) {
592  task->run(pc);
593  }
594  if constexpr (has_process<T>::value) {
595  AnalysisDataProcessorBuilder::invokeProcess(*(task.get()), pc.inputs(), &T::process);
596  }
597  std::apply([&pc](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::finalize(pc, x), ...); }, tupledTask);
598  };
599  }};
