115  void addNewPrecluster(UInt_t ip, UShort_t row)
116  {
117  mPreClusterHeads.push_back(mPixels.size());
118  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
119  mPixels.emplace_back(-1, ip);
120  int lastIndex = mPreClusterIndices.size();
121  mPreClusterIndices.push_back(lastIndex);
122  mCurr[row] = lastIndex; // store index of the new precluster in the current column buffer
123  }

126  void expandPreCluster(UInt_t ip, UShort_t row, int preClusIndex)
127  {
128  auto& firstIndex = mPreClusterHeads[mPreClusterIndices[preClusIndex]];
129  mPixels.emplace_back(firstIndex, ip);
