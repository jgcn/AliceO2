167 GPUdi() void GPUTPCGMPhysicalTrackModel::Set(const GPUTPCGMTrackParam& t)
168 {
169  float pti = CAMath::Abs(t.GetQPt());
170  if (pti < 1.e-4f) {
171  pti = 1.e-4f; // set 10000 GeV momentum for straight track
172  }
173  mQ = (t.GetQPt() >= 0) ? 1.f : -1.f; // only charged tracks are considered
174  mX = t.GetX();
175  mY = t.GetY();
176  mZ = t.GetZ();
177 
178  mPt = 1.f / pti;
179  mSinPhi = t.GetSinPhi();
180  if (mSinPhi > GPUCA_MAX_SIN_PHI) {
181  mSinPhi = GPUCA_MAX_SIN_PHI;
182  }
183  if (mSinPhi < -GPUCA_MAX_SIN_PHI) {
184  mSinPhi = -GPUCA_MAX_SIN_PHI;
185  }
186  mCosPhi = sqrt((1.f - mSinPhi) * (1.f + mSinPhi));
187  mSecPhi = 1.f / mCosPhi;
188  mDzDs = t.GetDzDs();
189  mDlDs = sqrt(1.f + mDzDs * mDzDs);
190  mP = mPt * mDlDs;
191 
192  mPy = mPt * mSinPhi;
193  mPx = mPt * mCosPhi;
194  mPz = mPt * mDzDs;
195  mQPt = mQ * pti;
196 }

210 GPUdi() void GPUTPCGMPhysicalTrackModel::UpdateValues()
211 {
212  float px = mPx;
213  if (CAMath::Abs(px) < 1.e-4f) {
214  px = copysign(1.e-4f, px);
215  }
216 
217  mPt = sqrt(px * px + mPy * mPy);
218  float pti = 1.f / mPt;
219  mP = sqrt(px * px + mPy * mPy + mPz * mPz);
220  mSinPhi = mPy * pti;
221  mCosPhi = px * pti;
222  mSecPhi = mPt / px;
223  mDzDs = mPz * pti;
224  mDlDs = mP * pti;
225  mQPt = mQ * pti;
226 }

246 GPUdi() float GPUTPCGMPhysicalTrackModel::GetMirroredY(float Bz) const
247 {
248  // get Y of the point which has the same X, but located on the other side of trajectory
249  if (CAMath::Abs(Bz) < 1.e-8f) {
250  Bz = 1.e-8f;
251  }
252  return mY - 2.f * mQ * mPx / Bz;
253 }

255 GPUdi() void GPUTPCGMPhysicalTrackModel::RotateLight(float alpha)
256 {
257  //* Rotate the coordinate system in XY on the angle alpha
258 
259  float cA = CAMath::Cos(alpha);
260  float sA = CAMath::Sin(alpha);
261  float x = mX, y = mY, px = mPx, py = mPy;
262  mX = x * cA + y * sA;
263  mY = -x * sA + y * cA;
264  mPx = px * cA + py * sA;
265  mPy = -px * sA + py * cA;
266 }
