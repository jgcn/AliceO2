98 void HalfDisk::createLadders()
99 {
100 
101  LOG(DEBUG1) << "CreateLadders: start building ladders";
102  for (Int_t iLadder = 0; iLadder < mSegmentation->getNLadders(); iLadder++) {
103 
104  LadderSegmentation* ladderSeg = mSegmentation->getLadder(iLadder);
105  if (!ladderSeg)
106  Fatal("CreateLadders", Form("No Segmentation found for ladder %d ", iLadder), 0, 0);
107  auto* ladder = new Ladder(ladderSeg);
108  TGeoVolume* ladVol = ladder->createVolume();
109 
110  // Position of the center on the ladder volume in the ladder coordinate system
111  TGeoBBox* shape = (TGeoBBox*)ladVol->GetShape();
112  Double_t center[3];
113  center[0] = shape->GetDX();
114  center[1] = shape->GetDY();
115  center[2] = shape->GetDZ();
116 
117  Double_t master[3];
118  ladderSeg->getTransformation()->LocalToMaster(center, master);
119  Int_t ladderId = Geometry::instance()->getLadderID(ladderSeg->GetUniqueID());
120 
121  mHalfDiskVolume->AddNode(ladVol, ladderId, new TGeoCombiTrans(master[0], master[1], master[2], ladderSeg->getTransformation()->GetRotation()));
122 
123  delete ladder;
124  }
