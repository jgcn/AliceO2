66 DataProcessorSpec CommonDataProcessors::getOutputObjSink(outputObjMap const& outMap)
67 {
68  auto writerFunction = [outMap](InitContext& ic) -> std::function<void(ProcessingContext&)> {
69  auto& callbacks = ic.services().get<CallbackService>();
70  auto outputObjects = std::make_shared<std::map<InputObjectRoute, InputObject>>();
71 
72  auto endofdatacb = [outputObjects](EndOfStreamContext& context) {
73  LOG(INFO) << "Writing merged objects to file";
74  std::string currentDirectory = "";
75  std::string currentFile = "";
76  TFile* f[OutputObjHandlingPolicy::numPolicies];
77  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
78  f[i] = nullptr;
79  }
80  for (auto& [route, entry] : *outputObjects) {
81  auto file = ROOTfileNames.find(route.policy);
82  if (file != ROOTfileNames.end()) {
83  auto filename = file->second;
84  if (f[route.policy] == nullptr) {
85  f[route.policy] = TFile::Open(filename.c_str(), "RECREATE");
86  }
87  auto nextDirectory = route.directory;
88  if ((nextDirectory != currentDirectory) || (filename != currentFile)) {
89  if (!f[route.policy]->FindKey(nextDirectory.c_str())) {
90  f[route.policy]->mkdir(nextDirectory.c_str());
91  }
92  currentDirectory = nextDirectory;
93  currentFile = filename;
94  }
95  (f[route.policy]->GetDirectory(currentDirectory.c_str()))->WriteObjectAny(entry.obj, entry.kind, entry.name.c_str());
96  }
97  }
98  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
99  if (f[i] != nullptr) {
100  f[i]->Close();
101  }
102  }
103  LOG(INFO) << "All outputs merged in their respective target files";
104  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
105  };
106 
107  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
108  return [outputObjects, outMap](ProcessingContext& pc) mutable -> void {
109  auto const& ref = pc.inputs().get("x");
110  if (!ref.header) {
111  LOG(ERROR) << "Header not found";
112  return;
113  }
114  if (!ref.payload) {
115  LOG(ERROR) << "Payload not found";
116  return;
117  }
118  auto datah = o2::header::get<o2::header::DataHeader*>(ref.header);
119  if (!datah) {
120  LOG(ERROR) << "No data header in stack";
121  return;
122  }
123 
124  auto objh = o2::header::get<o2::framework::OutputObjHeader*>(ref.header);
125  if (!objh) {
126  LOG(ERROR) << "No output object header in stack";
127  return;
128  }
129 
130  FairTMessage tm(const_cast<char*>(ref.payload), datah->payloadSize);
131  InputObject obj;
132  obj.kind = tm.GetClass();
133  if (obj.kind == nullptr) {
134  LOGP(error, "Cannot read class info from buffer.");
135  return;
136  }
137 
138  OutputObjHandlingPolicy policy = OutputObjHandlingPolicy::AnalysisObject;
139  if (objh)
140  policy = objh->mPolicy;
141 
142  obj.obj = tm.ReadObjectAny(obj.kind);
143  TNamed* named = static_cast<TNamed*>(obj.obj);
144  obj.name = named->GetName();
145  auto lookup = outMap.find(obj.name);
146  std::string directory{"VariousObjects"};
147  if (lookup != outMap.end()) {
148  directory = lookup->second;
149  }
150  InputObjectRoute key{obj.name, directory, policy};
151  auto existing = outputObjects->find(key);
152  if (existing == outputObjects->end()) {
153  outputObjects->insert(std::make_pair(key, obj));
154  return;
155  }
156  auto merger = existing->second.kind->GetMerge();
157  if (!merger) {
158  LOGP(error, "Already one object found for {}.", obj.name);
159  return;
160  }
161 
162  TList coll;
163  coll.Add(static_cast<TObject*>(obj.obj));
164  merger(existing->second.obj, &coll, nullptr);
165  };
166  };

178 DataProcessorSpec
179  CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
180  std::vector<InputSpec>& unmatched)
181 {
182  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
183  auto filename = ic.options().get<std::string>("outfile");
184  auto keepString = ic.options().get<std::string>("keep");
185 
186  if (filename.empty()) {
187  throw std::runtime_error("output file missing");
188  }
189 
190  bool hasOutputsToWrite = false;
191  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
192  VariableContext context;
193  for (auto& spec : danglingOutputInputs) {
194  auto concrete = DataSpecUtils::asConcreteDataMatcher(spec);
195  if (outputMatcher->match(concrete, context)) {
196  hasOutputsToWrite = true;
197  }
198  }
199  if (hasOutputsToWrite == false) {
200  return std::move([](ProcessingContext& pc) mutable -> void {
201  static bool once = false;
204  if (!once) {
205  LOG(INFO) << "No dangling output to be dumped.";
206  once = true;
207  }
208  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
209  });
210  }
211  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
212  return std::move([output, matcher = outputMatcher](ProcessingContext& pc) mutable -> void {
213  VariableContext matchingContext;
214  LOG(INFO) << "processing data set with " << pc.inputs().size() << " entries";
215  for (const auto& entry : pc.inputs()) {
216  LOG(INFO) << " " << *(entry.spec);
217  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
218  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
219  if (matcher->match(*header, matchingContext) == false) {
220  continue;
221  }
222  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
223  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
224  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
225  LOG(INFO) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
226  }
227  });
228  };
229 
230  std::vector<InputSpec> validBinaryInputs;
231  auto onlyTimeframe = [](InputSpec const& input) {
232  return input.lifetime == Lifetime::Timeframe;
233  };
234 
235  auto noTimeframe = [](InputSpec const& input) {
236  return input.lifetime != Lifetime::Timeframe;
237  };
238 
239  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
240  std::back_inserter(validBinaryInputs), onlyTimeframe);
241  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
242  std::back_inserter(unmatched), noTimeframe);
243 
244  DataProcessorSpec spec{
245  "internal-dpl-global-binary-file-sink",
246  validBinaryInputs,
247  Outputs{},
248  AlgorithmSpec(writerFunction),
249  {{"outfile", VariantType::String, "dpl-out.bin", {"Name of the output file"}},
250  {"keep", VariantType::String, "", {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile"}}}};
251 
252  return spec;
253 }
