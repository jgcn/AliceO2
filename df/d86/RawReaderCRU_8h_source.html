655 {
656  // std::cout << std::dec << s << " ";

660 inline void GBTFrame::updateSyncCheck(SyncArray& syncArray)
661 {
662  for (int s = 0; s < 5; s++)
663  for (int h = 0; h < 4; h++) {
664  // shift in a 1 if the halfword is 0x15
665  if (mFrameHalfWords[s][h] == 0x15)
666  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
667  // shift in a 0 if the halfword is 0xA
668  else if (mFrameHalfWords[s][h] == 0xA)
669  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
670  // otherwise reset the register to 0
671  else
672  mSyncCheckRegister[s] = 0;
673  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
674  // check the register content for the SYNC pattern
675  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
676  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
677  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
678  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
679  };
680  };
681 }

710 inline void GBTFrame::getFrameHalfWords()
711 {
712  uint32_t P[5][4] = {{19, 18, 17, 16}, {39, 38, 37, 36}, {63, 62, 61, 60}, {83, 82, 81, 80}, {107, 106, 105, 104}};
713  uint32_t res = 0;
714  // i = Stream, j = Halfword
715  for (int i = 0; i < 5; i++)
716  for (int j = 0; j < 4; j++) {
717  res = bit(P[i][j], 4) | bit(P[i][j] - 4, 3) | bit(P[i][j] - 8, 2) | bit(P[i][j] - 12, 1) | bit(P[i][j] - 16, 0);
718  mFrameHalfWords[i][j] = res;
719  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
720  };

734 inline void GBTFrame::getAdcValues(ADCRawData& rawData)
735 {
736  uint32_t pos;
737  uint32_t v0;
738  uint32_t v1;
739  // loop over all the 5 data streams
740  for (int s = 0; s < 5; s++) {
741  // assemble the 2 ADC words from 4 half-words. Which halfwords are
742  // used depends on the position of the SYNC pattern.
743  pos = mSyncPos[s].getHalfWordPosition();
744  if (pos == 0) {
745  v0 = (mPrevFrameHalfWords[s][2] << 5) | mPrevFrameHalfWords[s][1];
746  v1 = (mFrameHalfWords[s][0] << 5) | mPrevFrameHalfWords[s][3];
747  } else if (pos == 1) {
748  v0 = (mPrevFrameHalfWords[s][3] << 5) | mPrevFrameHalfWords[s][2];
749  v1 = (mFrameHalfWords[s][1] << 5) | mFrameHalfWords[s][0];
750  } else if (pos == 2) {
751  v0 = (mFrameHalfWords[s][0] << 5) | mPrevFrameHalfWords[s][3];
752  v1 = (mFrameHalfWords[s][2] << 5) | mFrameHalfWords[s][1];
753  } else if (pos == 3) {
754  v0 = (mFrameHalfWords[s][1] << 5) | mFrameHalfWords[s][0];
755  v1 = (mFrameHalfWords[s][3] << 5) | mFrameHalfWords[s][2];
756  };
757  // add the two ADC values for the current processed stream
758  if (mSyncPos[s].synched() == true) {
759  // std::cout << "Adding ADC" << std::endl;
760  rawData.add(s, v0, v1);
761  };
762  };
763  // std::cout << std::endl;
764 }
