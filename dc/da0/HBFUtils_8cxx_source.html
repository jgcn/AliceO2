19 int HBFUtils::getHBF(const IR& rec) const
20 {
22  auto diff = rec.differenceInBC(mFirstIR);
23  if (diff < 0) {
24  LOG(FATAL) << "IR " << rec.bc << '/' << rec.orbit << " is ahead of the reference IR "
25  << mFirstIR.bc << '/' << mFirstIR.orbit;
26  }
27  return diff / o2::constants::lhc::LHCMaxBunches;
28 }

31 int HBFUtils::fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const
32 {
33  // Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
34  // BCs between interaction records "fromIR" and "toIR" (inclusive).
35  dst.clear();
36  int hb0 = getHBF(fromIR), hb1 = getHBF(toIR);
37  if (fromIR.bc != mFirstIR.bc) { // unless we just starting the HBF of fromIR, it was already counted
38  hb0++;
39  }
40  for (int ihb = hb0; ihb <= hb1; ihb++) {
41  dst.emplace_back(getIRHBF(ihb));
42  }
43  return dst.size();
44 }

92 void HBFUtils::dumpRDH(const o2::header::RAWDataHeaderV5& rdh)
93 {
94  constexpr int hsz = sizeof(o2::header::RAWDataHeaderV5);
95  static_assert(hsz == 64, "Expect 64 bytes long header");
96  const uint32_t* w32 = reinterpret_cast<const uint32_t*>(&rdh);
97  for (int iw128 = 0; iw128 < hsz / 16; iw128++) {
98  printf("[rdh%d] 0x:", iw128);
99  for (int iw = hsz / sizeof(uint32_t); iw--;) {
100  printf(" %08x", w32[iw128 * 16 / sizeof(uint32_t) + iw]);
101  }
102  printf("\n");
103  }
104 }
