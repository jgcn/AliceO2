32 int Decoder::next(unsigned int nBits)
33 {
35  int value = 0;
36  for (int ibit = 0; ibit < nBits; ++ibit) {
37  if (mBitIndex == raw::sElementSizeInBits) {
38  mBitIndex = 0;
39  ++mVectorIndex;
40  nextPayload();
41  }
42  bool isOn = (mBytes[mVectorIndex] >> mBitIndex) & 0x1;
43  if (isOn) {
44  value |= (1 << ibit);
45  }
46  ++mBitIndex;
47  }
48  return value;
49 }

81 bool Decoder::nextPayload()
82 {
84  while (mVectorIndex == mNextHeaderIndex) {
85  mRDH = reinterpret_cast<const header::RAWDataHeader*>(&mBytes[mVectorIndex]);
86  mHeaderIndex = mNextHeaderIndex;
87  mNextHeaderIndex += mRDH->offsetToNext / raw::sElementSizeInBytes;
88  mBitIndex = 0;
89  if (mRDH->memorySize > raw::sHeaderSizeInBytes) {
90  // Payload is not empty: go to end of header
91  mVectorIndex += raw::sHeaderSizeInElements;
92  } else {
93  // Payload is empty: go directly to next header and check it
94  mVectorIndex = mNextHeaderIndex;
95  if (isEndOfBuffer()) {
96  return false;
97  }
98  }
99  }
100  return true;
101 }

103 bool Decoder::nextColumnData()
104 {
106  if (isEndOfBuffer() || !nextPayload()) {
107  return false;
108  }
109 
110  EventType eventType = static_cast<EventType>(next(sNBitsEventType));
111  uint16_t localClock = next(sNBitsLocalClock);
112  InteractionRecord intRec(localClock, mRDH->triggerOrbit);
113  auto firstEntry = mData.size();
114  int nFiredRPCs = next(sNBitsNFiredRPCs);
115  ColumnData colData;
116  for (int irpc = 0; irpc < nFiredRPCs; ++irpc) {
117  int rpcId = next(sNBitsRPCId);
118  int nFiredColumns = next(sNBitsNFiredColumns);
119  colData.deId = static_cast<uint8_t>(rpcId);
120  for (int icol = 0; icol < nFiredColumns; ++icol) {
121  colData.columnId = static_cast<uint8_t>(next(sNBitsColumnId));
122  for (int iline = 0; iline < 4; ++iline) {
123  // Reset pattern
124  colData.setBendPattern(0, iline);
125  }
126  int nFiredBoards = next(sNBitsNFiredBoards);
127  for (int iboard = 0; iboard < nFiredBoards; ++iboard) {
128  int boardId = next(sNBitsBoardId);
129  colData.setNonBendPattern(next(sNBitsFiredStrips));
130  colData.setBendPattern(next(sNBitsFiredStrips), boardId);
131  }
132  mData.emplace_back(colData);
133  }
134  }
135 
136  mROFRecords.emplace_back(intRec, eventType, firstEntry, mData.size() - firstEntry);
137 
138  // The payload bits are contiguous.
139  // However there is an overhead between the last payload byte and the next header.
140  // If we read the full memory size, we
141  // Which means that the size is too small to have another block of data.
142  // When we are close to the new header, we jump directly to the next header
143  if ((mVectorIndex - mHeaderIndex + 1) * raw::sElementSizeInBytes == mRDH->memorySize) {
144  mVectorIndex = mNextHeaderIndex;
145  mBitIndex = 0;
146  }
147 
148  return true;
149 }
