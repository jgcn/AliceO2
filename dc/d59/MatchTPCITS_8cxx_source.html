258 //______________________________________________
259 bool MatchTPCITS::validateTPCMatch(int mtID)
260 {
261  auto& tpcMatch = mMatchesTPC[mtID];
262  auto& rcTPC = mMatchRecordsTPC[tpcMatch.first]; // best TPC->ITS match
263  if (rcTPC.nextRecID == Validated)
264  return false; // RS do we need this
265  // check if it is consistent with corresponding ITS->TPC match
266  auto& itsMatch = mMatchesITS[rcTPC.matchID]; // matchCand of partner ITS track
267  auto& rcITS = mMatchRecordsITS[itsMatch.first]; // best ITS->TPC match
268  if (rcTPC.nextRecID == Validated)
269  return false; // RS do we need this ?
270  if (rcITS.matchID == mtID) { // is best matching TPC track for this ITS track actually mtID?
271 
272  // unlink winner TPC track from all ITS candidates except winning one
273  int nextTPC = rcTPC.nextRecID;
274  while (nextTPC > MinusOne) {
275  auto& rcTPCrem = mMatchRecordsTPC[nextTPC];
276  removeTPCfromITS(mtID, rcTPCrem.matchID); // remove references on mtID from ITS match=rcTPCrem.matchID
277  nextTPC = rcTPCrem.nextRecID;
278  }
279  rcTPC.nextRecID = Validated;
280  int itsWinID = rcTPC.matchID;
281 
282  // unlink winner ITS match from all TPC matches using it
283  int nextITS = rcITS.nextRecID;
284  while (nextITS > MinusOne) {
285  auto& rcITSrem = mMatchRecordsITS[nextITS];
286  removeITSfromTPC(itsWinID, rcITSrem.matchID); // remove references on itsWinID from TPC match=rcITSrem.matchID
287  nextITS = rcITSrem.nextRecID;
288  }
289  rcITS.nextRecID = Validated;
290  return true;
291  }
292  return false;
293 }

417 bool MatchTPCITS::prepareTPCTracks()
418 {
420  mMatchRecordsTPC.clear();
421  int curTracksEntry = 0;
422 
423  if (!mDPLIO) { // in the DPL IO mode the input TPC tracks must be already attached
424  if (!loadTPCTracksNextChunk()) {
425  return false;
426  }
427  curTracksEntry = mTreeTPCTracks ? mTreeTPCTracks->GetReadEntry() : 0;
428  }
429 
430  int ntr = mTPCTracksArrayInp.size();
431  mMatchesTPC.reserve(mMatchesTPC.size() + ntr);
432  // number of records might be actually more than N tracks!
433  mMatchRecordsTPC.reserve(mMatchRecordsTPC.size() + mMaxMatchCandidates * ntr);
434 
435  // copy the track params, propagate to reference X and build sector tables
436  mTPCWork.clear();
437  mTPCWork.reserve(ntr);
438  if (mMCTruthON) {
439  mTPCLblWork.clear();
440  mTPCLblWork.reserve(ntr);
441  }
442  for (int sec = o2::constants::math::NSectors; sec--;) {
443  mTPCSectIndexCache[sec].clear();
444  mTPCSectIndexCache[sec].reserve(100 + 1.2 * ntr / o2::constants::math::NSectors);
445  mTPCTimeBinStart[sec].clear();
446  }
447 
448  for (int it = 0; it < ntr; it++) {
449 
450  const auto& trcOrig = mTPCTracksArrayInp[it];
451 
452  // make sure the track was propagated to inner TPC radius at the ref. radius
453  if (trcOrig.getX() > XTPCInnerRef + 0.1)
454  continue; // failed propagation to inner TPC radius, cannot be matched
455 
456  // create working copy of track param
457  mTPCWork.emplace_back(static_cast<const o2::track::TrackParCov&>(trcOrig), curTracksEntry, it);
458  auto& trc = mTPCWork.back();
459  // propagate to matching Xref
460  if (!propagateToRefX(trc)) {
461  mTPCWork.pop_back(); // discard track whose propagation to XMatchingRef failed
462  continue;
463  }
464  if (mMCTruthON) {
465  mTPCLblWork.emplace_back(mTPCTrkLabels->getLabels(it)[0]);
466  }
467 
468  float time0 = trcOrig.getTime0() - mNTPCBinsFullDrift;
469  trc.timeBins.set(time0 - trcOrig.getDeltaTBwd() - mTPCTimeEdgeTSafeMargin,
470  time0 + trcOrig.getDeltaTFwd() + mTPCTimeEdgeTSafeMargin);
471  // assign min max possible Z for this track which still respects the clusters A/C side
472  if (trcOrig.hasASideClustersOnly()) {
473  trc.zMin = trc.getZ() - trcOrig.getDeltaTBwd() * mTPCBin2Z;
474  trc.zMax = trc.getZ() + trcOrig.getDeltaTFwd() * mTPCBin2Z;
475  } else if (trcOrig.hasCSideClustersOnly()) {
476  trc.zMin = trc.getZ() - trcOrig.getDeltaTFwd() * mTPCBin2Z;
477  trc.zMax = trc.getZ() + trcOrig.getDeltaTBwd() * mTPCBin2Z;
478  }
479  // TODO : special treatment of tracks crossing the CE
480 
481  // cache work track index
482  mTPCSectIndexCache[o2::utils::Angle2Sector(trc.getAlpha())].push_back(mTPCWork.size() - 1);
483  }
484  float maxTimeBin = 0;
485  int nITSROFs = mITSROFTimes.size();
486  // sort tracks in each sector according to their timeMax
487  for (int sec = o2::constants::math::NSectors; sec--;) {
488  auto& indexCache = mTPCSectIndexCache[sec];
489  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " TPC tracks";
490  if (!indexCache.size())
491  continue;
492  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
493  auto& trcA = mTPCWork[a];
494  auto& trcB = mTPCWork[b];
495  return (trcA.timeBins.tmax - trcB.timeBins.tmax) < 0.;
496  });
497 
498  // build array of 1st entries with tmax corresponding to each ITS ROF (or trigger)
499  float tmax = mTPCWork[indexCache.back()].timeBins.tmax;
500  if (maxTimeBin < tmax) {
501  maxTimeBin = tmax;
502  }
503  int nbins = 1 + tpcTimeBin2ITSROFrame(tmax);
504  auto& tbinStart = mTPCTimeBinStart[sec];
505  tbinStart.resize(nbins, -1);
506  int itsROF = 0;
507  tbinStart[0] = itsROF;
508  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
509  auto& trc = mTPCWork[indexCache[itr]];
510 
511  while (trc.timeBins.tmax < mITSROFTimes[itsROF].tmin && itsROF < nITSROFs) {
512  itsROF++;
513  }
514  if (tbinStart[itsROF] == -1) {
515  tbinStart[itsROF] = itr;
516  }
517  }
518  for (int i = 1; i < nbins; i++) {
519  if (tbinStart[i] == -1) { // fill gaps with preceding indices
520  tbinStart[i] = tbinStart[i - 1];
521  }
522  }
523  } // loop over tracks of single sector
524 
525  // create mapping from TPC time-bins to ITS ROFs
526 
527  if (maxTimeBin < mITSROFTimes.back().tmax) {
528  maxTimeBin = mITSROFTimes.back().tmax;
529  }
530  int nb = int(maxTimeBin) + 1;
531  mITSROFofTPCBin.resize(nb, -1);
532  int itsROF = 0;
533  for (int ib = 0; ib < nb; ib++) {
534  while (ib < mITSROFTimes[itsROF].tmin && itsROF < nITSROFs) {
535  itsROF++;
536  }
537  mITSROFofTPCBin[ib] = itsROF;
538  }
539  return true;
540 }

560  LOG(INFO) << "Empty TF";
561  return false;
562  }
563 
564  mITSWork.clear();
565  mITSROFTimes.clear();
566  // number of records might be actually more than N tracks!
567  mMatchRecordsITS.clear(); // RS TODO reserve(mMatchRecordsITS.size() + mMaxMatchCandidates*ntr);
568  if (mMCTruthON) {
569  mITSLblWork.clear();
570  }
571 
572  for (int sec = o2::constants::math::NSectors; sec--;) {
573  mITSSectIndexCache[sec].clear();
574  mITSTimeBinStart[sec].clear();
575  mITSTimeBinStart[sec].resize(nROFs, -1); // start of ITS work tracks in every sector
576  }
577  setStartIR(mITSTrackROFRec[0].getBCData());
578  for (int irof = 0; irof < nROFs; irof++) {
579  const auto& rofRec = mITSTrackROFRec[irof];
580  // in case of the input from the tree make sure needed entry is loaded
581  auto rEntry = rofRec.getROFEntry().getEvent();
582  if (!mDPLIO && mCurrITSTracksTreeEntry != rEntry) { // in DPL IO mode the input tracks must be already attached
583  mTreeITSTracks->GetEntry((mCurrITSTracksTreeEntry = rEntry));
584  mITSTracksArrayInp = gsl::span<const o2::its::TrackITS>(mITSTracksArrayPtr->data(), mITSTracksArrayPtr->size());
585  mITSTrackClusIdxInp = gsl::span<const int>(mITSTrackClusIdxPtr->data(), mITSTrackClusIdxPtr->size());
586  }
587  float tmn = intRecord2TPCTimeBin(rofRec.getBCData()); // ITS track min time in TPC time-bins
588  mITSROFTimes.emplace_back(tmn, tmn + mITSROFrame2TPCBin); // ITS track min/max time in TPC time-bins
589 
590  for (int sec = o2::constants::math::NSectors; sec--;) { // start of sector's tracks for this ROF
591  mITSTimeBinStart[sec][irof] = mITSSectIndexCache[sec].size(); // The sorting does not affect this
592  }
593 
594  int trlim = rofRec.getROFEntry().getIndex() + rofRec.getNROFEntries();
595  for (int it = rofRec.getROFEntry().getIndex(); it < trlim; it++) {
596  const auto& trcOrig = mITSTracksArrayInp[it];
597  if (trcOrig.getParamOut().getX() < 1.) {
598  continue; // backward refit failed
599  }
600  int nWorkTracks = mITSWork.size();
601  // working copy of outer track param
602  mITSWork.emplace_back(static_cast<const o2::track::TrackParCov&>(trcOrig.getParamOut()), rEntry, it);
603  auto& trc = mITSWork.back();
604 
605  if (!trc.rotate(o2::utils::Angle2Alpha(trc.getPhiPos()))) {
606  mITSWork.pop_back(); // discard failed track
607  continue;
608  }
609  // make sure the track is at the ref. radius
610  if (!propagateToRefX(trc)) {
611  mITSWork.pop_back(); // discard failed track
612  continue; // add to cache only those ITS tracks which reached ref.X and have reasonable snp
613  }
614  if (mMCTruthON) {
615  mITSLblWork.emplace_back(mITSTrkLabels->getLabels(it)[0]);
616  }
617  trc.roFrame = irof;
618 
619  // cache work track index
620  int sector = o2::utils::Angle2Sector(trc.getAlpha());
621  mITSSectIndexCache[sector].push_back(nWorkTracks);
622 
623  // If the ITS track is very close to the sector edge, it may match also to a TPC track in the neighb. sector.
624  // For a track with Yr and Phir at Xr the distance^2 between the poisition of this track in the neighb. sector
625  // when propagated to Xr (in this neighbouring sector) and the edge will be (neglecting the curvature)
626  // [(Xr*tg(10)-Yr)/(tgPhir+tg70)]^2 / cos(70)^2 // for the next sector
627  // [(Xr*tg(10)+Yr)/(tgPhir-tg70)]^2 / cos(70)^2 // for the prev sector
628  // Distances to the sector edges in neighbourings sectors (at Xref in theit proper frames)
629  float tgp = trc.getSnp();
630  tgp /= std::sqrt((1.f - tgp) * (1.f + tgp)); // tan of track direction XY
631 
632  // sector up
633  float dy2Up = (YMaxAtXMatchingRef - trc.getY()) / (tgp + Tan70);
634  if ((dy2Up * dy2Up * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector up
635  addTrackCloneForNeighbourSector(trc, sector < (o2::constants::math::NSectors - 1) ? sector + 1 : 0);
636  }
637  // sector down
638  float dy2Dn = (YMaxAtXMatchingRef + trc.getY()) / (tgp - Tan70);
639  if ((dy2Dn * dy2Dn * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector down
640  addTrackCloneForNeighbourSector(trc, sector > 1 ? sector - 1 : o2::constants::math::NSectors - 1);
641  }
642  }
643  }
644 
645  // sort tracks in each sector according to their time, then tgl
646  for (int sec = o2::constants::math::NSectors; sec--;) {
647  auto& indexCache = mITSSectIndexCache[sec];
648  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " ITS tracks";
649  if (!indexCache.size()) {
650  continue;
651  }
652  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
653  auto& trackA = mITSWork[a];
654  auto& trackB = mITSWork[b];
655  if (trackA.roFrame < trackB.roFrame) { // ITS tracks have the same time coverage
656  return true;
657  } else if (trackA.roFrame > trackB.roFrame) {
658  return false;
659  }
660  return trackA.getTgl() < trackB.getTgl();
661  });
662 
663  /* TOREMOVE?
664  // build array of 1st entries with of each ITS ROFrame
665  int nbins = 1 + mITSWork[indexCache.back()].roFrame;
666  auto& tbinStart = mITSTimeBinStart[sec];
667  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
668  tbinStart[0] = 0;
669  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
670  auto& trc = mITSWork[indexCache[itr]];
671  if (tbinStart[trc.roFrame] == -1) {
672  tbinStart[trc.roFrame] = itr;
673  }
674  }
675  for (int i = 1; i < nbins; i++) {
676  if (tbinStart[i] == -1) { // fill gaps with preceding indices. TODO ? Shall we do this for cont.readout in ITS only?
677  tbinStart[i] = tbinStart[i - 1];
678  }
679  }
680  */
681  } // loop over tracks of single sector
682  mMatchesITS.reserve(mITSWork.size());
683  mMatchRecordsITS.reserve(mITSWork.size() * mMaxMatchCandidates);
684 
685  return true;
686 }

726 //_____________________________________________________
727 void MatchTPCITS::doMatching(int sec)
728 {
730  auto& cacheITS = mITSSectIndexCache[sec]; // array of cached ITS track indices for this sector
731  auto& cacheTPC = mTPCSectIndexCache[sec]; // array of cached ITS track indices for this sector
732  auto& tbinStartTPC = mTPCTimeBinStart[sec]; // array of 1st TPC track with timeMax in ITS ROFrame
733  auto& tbinStartITS = mITSTimeBinStart[sec];
734  int nTracksTPC = cacheTPC.size(), nTracksITS = cacheITS.size();
735  if (!nTracksTPC || !nTracksITS) {
736  LOG(INFO) << "Matchng sector " << sec << " : N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS << " in sector "
737  << sec;
738  return;
739  }
740 
742  float maxTDriftSafe = (mNTPCBinsFullDrift + mTPCITSTimeBinSafeMargin + mTPCTimeEdgeTSafeMargin);
743 
744  // get min ROFrame (in TPC time-bins) of ITS tracks currently in cache
745  auto minROFITS = mITSWork[cacheITS.front()].roFrame;
746 
747  if (minROFITS >= int(tbinStartTPC.size())) {
748  LOG(INFO) << "ITS min ROFrame " << minROFITS << " exceeds all cached TPC track ROF eqiuvalent "
749  << cacheTPC.size() - 1;
750  return;
751  }
752 
753  int nCheckTPCControl = 0, nCheckITSControl = 0, nMatchesControl = 0; // temporary
754 
755  int idxMinTPC = tbinStartTPC[minROFITS]; // index of 1st cached TPC track within cached ITS ROFrames
756  for (int itpc = idxMinTPC; itpc < nTracksTPC; itpc++) {
757  auto& trefTPC = mTPCWork[cacheTPC[itpc]];
758  // estimate ITS 1st ROframe bin this track may match to: TPC track are sorted according to their
759  // timeMax, hence the timeMax - MaxmNTPCBinsFullDrift are non-decreasing
760  int itsROBin = tpcTimeBin2ITSROFrame(trefTPC.timeBins.tmax - maxTDriftSafe);
761  if (itsROBin >= int(tbinStartITS.size())) { // time of TPC track exceeds the max time of ITS in the cache
762  break;
763  }
764  int iits0 = tbinStartITS[itsROBin];
765  nCheckTPCControl++;
766  for (auto iits = iits0; iits < nTracksITS; iits++) {
767  auto& trefITS = mITSWork[cacheITS[iits]];
768  const auto& timeITS = mITSROFTimes[trefITS.roFrame];
769  // compare if the ITS and TPC tracks may overlap in time
770  if (trefTPC.timeBins.tmax < timeITS.tmin) {
771  // since TPC tracks are sorted in timeMax and ITS tracks are sorted in timeMin
772  // all following ITS tracks also will not match
773  break;
774  }
775  if (trefTPC.timeBins.tmin > timeITS.tmax) { // its bracket is fully before TPC bracket
776  continue;
777  }
778  nCheckITSControl++;
779  float chi2 = -1;
780  int rejFlag = compareTPCITSTracks(trefITS, trefTPC, chi2);
781 
782 #ifdef _ALLOW_DEBUG_TREES_
783  if (mDBGOut && ((rejFlag == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
784  fillTPCITSmatchTree(cacheITS[iits], cacheTPC[itpc], rejFlag, chi2);
785  }
786 #endif
787 
788  if (rejFlag == RejectOnTgl) {
789  // ITS tracks in each ROFrame are ordered in Tgl, hence if this check failed on Tgl check
790  // (i.e. tgl_its>tgl_tpc+tolerance), tnem all other ITS tracks in this ROFrame will also have tgl too large.
791  // Jump on the 1st ITS track of the next ROFrame
792  int rof = trefITS.roFrame;
793  bool stop = false;
794  do {
795  if (++rof >= int(tbinStartITS.size())) {
796  stop = true;
797  break; // no more ITS ROFrames in cache
798  }
799  iits = tbinStartITS[rof] - 1; // next track to be checked -1
800  } while (iits <= tbinStartITS[trefITS.roFrame]); // skip empty bins
801  if (stop) {
802  break;
803  }
804  continue;
805  }
806  if (rejFlag != Accept) {
807  continue;
808  }
809  mTimerReg.Start(false);
810  registerMatchRecordTPC(trefITS, trefTPC, chi2); // register matching candidate
811  mTimerReg.Stop();
812  nMatchesControl++;
813  }
814  }
815 
816  LOG(INFO) << "Match sector " << sec << " N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS
817  << " N TPC tracks checked: " << nCheckTPCControl << " (starting from " << idxMinTPC
818  << "), checks: " << nCheckITSControl << ", matches:" << nMatchesControl;
819 }

843 //______________________________________________
844 bool MatchTPCITS::registerMatchRecordTPC(TrackLocITS& tITS, TrackLocTPC& tTPC, float chi2)
845 {
848 
849  auto& mtcTPC = getTPCMatchEntry(tTPC); // get matchCand structure of this TPC track, create if none
850  int nextID = mtcTPC.first; // get 1st matchRecord this matchCand refers to
851  if (nextID < 0) { // no matches yet, just add new record
852  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
853  mtcTPC.first = mMatchRecordsTPC.size(); // new record will be added in the end
854  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2); // create new record with empty reference on next match
855  return true;
856  }
857 
858  int count = 0, topID = MinusOne;
859  do {
860  auto& nextMatchRec = mMatchRecordsTPC[nextID];
861  count++;
862  if (chi2 < nextMatchRec.chi2) { // need to insert new record before nextMatchRec?
863  if (count < mMaxMatchCandidates) {
864  break; // will insert in front of nextID
865  } else { // max number of candidates reached, will overwrite the last one
866  nextMatchRec.chi2 = chi2;
867  suppressMatchRecordITS(nextMatchRec.matchID, tTPC.matchID); // flag as disabled the overriden ITS match
868  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
869  nextMatchRec.matchID = tITS.matchID; // reuse the record of suppressed ITS match to store better one
870  return true;
871  }
872  }
873  topID = nextID; // check next match record
874  nextID = nextMatchRec.nextRecID;
875  } while (nextID > MinusOne);
876 
877  // if count == mMaxMatchCandidates, the max number of candidates was already reached, and the
878  // new candidated was either discarded (if its chi2 is worst one) or has overwritten worst
879  // existing candidate. Otherwise, we need to add new entry
880  if (count < mMaxMatchCandidates) {
881  if (topID < 0) { // the new match is top candidate
882  topID = mtcTPC.first = mMatchRecordsTPC.size(); // register new record as top one
883  } else { // there are better candidates
884  topID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC.size(); // register to his parent
885  }
886  // nextID==-1 will mean that the while loop run over all candidates->the new one is the worst (goes to the end)
887  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
888  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2, nextID); // create new record with empty reference on next match
889  // make sure that after addition the number of candidates don't exceed allowed number
890  count++;
891  while (nextID > MinusOne) {
892  if (count > mMaxMatchCandidates) {
893  suppressMatchRecordITS(mMatchRecordsTPC[nextID].matchID, tTPC.matchID);
894  // exclude nextID record, w/o changing topID (which becomes the last record)
895  nextID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC[nextID].nextRecID;
896  continue;
897  }
898  count++;
899  topID = nextID;
900  nextID = mMatchRecordsTPC[nextID].nextRecID;
901  }
902  return true;
903  } else {
904  return false; // unless nextID was assigned OverrideExisting, new candidate was discarded
905  }
906 }

909 void MatchTPCITS::registerMatchRecordITS(TrackLocITS& tITS, int matchTPCID, float chi2)
910 {
912  auto& itsMatch = getITSMatchEntry(tITS); // if needed, create new entry
913  int nextRecord = itsMatch.first; // entry of 1st match record in mMatchRecordsITS

919  }

934  }
935  topID = nextRecord;
936  nextRecord = mMatchRecordsITS[nextRecord].nextRecID;
937  } while (nextRecord > MinusOne);
938

940  mMatchRecordsITS[topID].nextRecID = idnew; // register new link
941 }

943 //______________________________________________
944 int MatchTPCITS::compareTPCITSTracks(const TrackLocITS& tITS, const TrackLocTPC& tTPC, float& chi2) const
945 {
947  auto& trackTPC = tTPC;
948  auto& trackITS = tITS;
949  chi2 = -1.f;
950  int rejFlag = Accept;
951  float diff; // make rough check differences and their nsigmas
952 
953  // start with check on Tgl, since rjection on it will allow to profit from sorting
954  diff = trackITS.getParam(o2::track::kTgl) - trackTPC.getParam(o2::track::kTgl);
955  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kTgl], RejectOnTgl))) {
956  return rejFlag;
957  }
958  diff *= diff / (trackITS.getDiagError2(o2::track::kTgl) + trackTPC.getDiagError2(o2::track::kTgl));
959  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kTgl], RejectOnTgl + NSigmaShift))) {
960  return rejFlag;
961  }
962 
963  diff = trackITS.getParam(o2::track::kY) - trackTPC.getParam(o2::track::kY);
964  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kY], RejectOnY))) {
965  return rejFlag;
966  }
967  diff *= diff / (trackITS.getDiagError2(o2::track::kY) + trackTPC.getDiagError2(o2::track::kY));
968  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kY], RejectOnY + NSigmaShift))) {
969  return rejFlag;
970  }
971 
972  if (mCompareTracksDZ) { // in continuous mode we usually don't use DZ
973  diff = trackITS.getParam(o2::track::kZ) - trackTPC.getParam(o2::track::kZ);
974  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kZ], RejectOnZ))) {
975  return rejFlag;
976  }
977  diff *= diff / (trackITS.getDiagError2(o2::track::kZ) + trackTPC.getDiagError2(o2::track::kZ));
978  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kZ], RejectOnZ + NSigmaShift))) {
979  return rejFlag;
980  }
981  } else { // in continuous mode we use special check on allowed Z range
982  if (trackITS.getParam(o2::track::kZ) - tTPC.zMax > mCrudeAbsDiffCut[o2::track::kZ])
983  return RejectOnZ;
984  if (trackITS.getParam(o2::track::kZ) - tTPC.zMin < -mCrudeAbsDiffCut[o2::track::kZ])
985  return -RejectOnZ;
986  }
987 
988  diff = trackITS.getParam(o2::track::kSnp) - trackTPC.getParam(o2::track::kSnp);
989  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kSnp], RejectOnSnp))) {
990  return rejFlag;
991  }
992  diff *= diff / (trackITS.getDiagError2(o2::track::kSnp) + trackTPC.getDiagError2(o2::track::kSnp));
993  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kSnp], RejectOnSnp + NSigmaShift))) {
994  return rejFlag;
995  }
996 
997  diff = trackITS.getParam(o2::track::kQ2Pt) - trackTPC.getParam(o2::track::kQ2Pt);
998  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kQ2Pt], RejectOnQ2Pt))) {
999  return rejFlag;
1000  }
1001  diff *= diff / (trackITS.getDiagError2(o2::track::kQ2Pt) + trackTPC.getDiagError2(o2::track::kQ2Pt));
1002  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kQ2Pt], RejectOnQ2Pt + NSigmaShift))) {
1003  return rejFlag;
1004  }
1005  // calculate mutual chi2 excluding Z in continuos mode
1006  chi2 = getPredictedChi2NoZ(tITS, tTPC);
1007  if (chi2 > mCutMatchingChi2)
1008  return RejectOnChi2;
1009 
1010  return Accept;
1011 }
