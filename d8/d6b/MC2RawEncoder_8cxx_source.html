16 template <class Mapping>
17 void MC2RawEncoder<Mapping>::init()
18 {
19  // limit RUs to convert to existing ones
20  mRUSWMax = (mRUSWMax < uint8_t(mMAP.getNRUs())) ? mRUSWMax : mMAP.getNRUs() - 1;
21  //
22  mNLinks = 0;
23  for (uint8_t ru = mRUSWMin; ru <= mRUSWMax; ru++) {
24  auto& ruData = getCreateRUDecode(ru);
25  int nLinks = 0;
26  for (int il = 0; il < MaxLinksPerRU; il++) {
27  if (ruData.links[il]) {
28  nLinks++;
29  if (ruData.links[il]->packetCounter < 0) {
30  ruData.links[il]->packetCounter = 0; // reset only once
31  }
32  }
33  }
34  mNLinks += nLinks;
35  if (!nLinks) {
36  LOG(INFO) << "Imposing single link readout for RU " << int(ru);
37  ruData.links[0] = std::make_unique<GBTLink>();
38  ruData.links[0]->lanes = mMAP.getCablesOnRUType(ruData.ruInfo->ruType);
39  ruData.links[0]->id = 0;
40  ruData.links[0]->cruID = ru;
41  ruData.links[0]->feeID = mMAP.RUSW2FEEId(ruData.ruInfo->idSW, 0);
42  ruData.links[0]->packetCounter = 0;
43  mNLinks++;
44  }
45  }
46 
47  mLastIR = mHBFUtils.getFirstIR(); // TFs are counted from this IR
48 
49  assert(mNLinks > 0);
50 }

53 template <class Mapping>
54 void MC2RawEncoder<Mapping>::finalize()
55 {
56  // close open HBFs, write empty HBFs until the end of the TF and close all streams
57 
58  int tf = mHBFUtils.getTF(mCurrIR);
59  mCurrIR = mHBFUtils.getIRTF(tf + 1) - 1; // last IR of the current TF
60  mHBFUtils.fillHBIRvector(mHBIRVec, mLastIR, mCurrIR);
61  for (const auto& ir : mHBIRVec) {
62  mLastIR = ir;
63  mRDH = mHBFUtils.createRDH<RDH>(mLastIR);
64  openHBF(); // open new HBF for all links
65  }
66  mLastIR++; // next call to fillHBIRvector should start from yet non-processed IR
67  closeHBF(); // close all HBFs
68  flushAllLinks(); // write data remaining in the link
69 }

72 template <class Mapping>
73 void MC2RawEncoder<Mapping>::digits2raw(gsl::span<const Digit> digits, const o2::InteractionRecord& bcData)
74 {
75  // Convert digits to raw data
76  // The digits in the span/vector must be in increasing chipID order
77  // Return the number of pages in the link with smallest amount of pages
78 
79  if (!mNLinks) {
80  init();
81  }
82 
83  int nDigTot = digits.size();
84  mCurrIR = bcData;
85 
86  // get list of IR's for which HBF should be generated
87  mHBFUtils.fillHBIRvector(mHBIRVec, mLastIR, mCurrIR);
88  for (const auto& ir : mHBIRVec) {
89  mLastIR = ir;
90  mRDH = mHBFUtils.createRDH<RDH>(mLastIR);
91  openHBF(); // open new HBF for all links
92  }
93  mLastIR++; // next call to fillHBIRvector should start from yet non-processed IR
94 
95  // place digits into corresponding chip buffers
96  ChipPixelData* curChipData = nullptr;
97  ChipInfo chInfo;
98  UShort_t curChipID = 0xffff; // currently processed SW chip id
99  for (const auto& dig : digits) {
100  if (curChipID != dig.getChipIndex()) {
101  mMAP.getChipInfoSW(dig.getChipIndex(), chInfo);
102  if (chInfo.ru < mRUSWMin || chInfo.ru > mRUSWMax) { // ignore this chip?
103  continue;
104  }
105  auto* ru = getRUDecode(chInfo.ru);
106  curChipID = dig.getChipIndex();
107  curChipData = &ru->chipsData[ru->nChipsFired++];
108  curChipData->setChipID(chInfo.chOnRU->id); // set chip ID within the RU
109  }
110  curChipData->getData().emplace_back(&dig); // add new digit to the container
111  }
112 
113  // convert digits to alpide data in the per-cable buffers
114  for (int iru = int(mRUSWMin); iru <= int(mRUSWMax); iru++) {
115  auto& ru = *getRUDecode(iru);
116  uint16_t next2Proc = 0, nchTot = mMAP.getNChipsOnRUType(ru.ruInfo->ruType);
117  for (int ich = 0; ich < ru.nChipsFired; ich++) {
118  auto& chipData = ru.chipsData[ich];
119  convertEmptyChips(next2Proc, chipData.getChipID(), ru); // if needed store EmptyChip flags for the empty chips
120  next2Proc = chipData.getChipID() + 1;
121  convertChip(chipData, ru);
122  chipData.clear();
123  }
124  convertEmptyChips(next2Proc, nchTot, ru); // if needed store EmptyChip flags
125  fillGBTLinks(ru); // flush per-lane buffers to link buffers
126  }
127 }

130 template <class Mapping>
131 void MC2RawEncoder<Mapping>::convertChip(ChipPixelData& chipData, RUDecodeData& ru)
132 {
134  const auto& chip = *mMAP.getChipOnRUInfo(ru.ruInfo->ruType, chipData.getChipID());
135  ru.cableHWID[chip.cableSW] = chip.cableHW; // register the cable HW ID
136  auto& pixels = chipData.getData();
137  std::sort(pixels.begin(), pixels.end(),
138  [](auto lhs, auto rhs) {
139  return (lhs.getRow() < rhs.getRow()) ? true : ((lhs.getRow() > rhs.getRow()) ? false : (lhs.getCol() < rhs.getCol()));
140  });
141  ru.cableData[chip.cableSW].ensureFreeCapacity(40 * (2 + pixels.size())); // make sure buffer has enough capacity
142  mCoder.encodeChip(ru.cableData[chip.cableSW], chipData, chip.chipOnModuleHW, mCurrIR.bc);
143 }

146 template <class Mapping>
147 void MC2RawEncoder<Mapping>::openPageLinkHBF(GBTLink& link)
148 {
150  if (mRDH.triggerType & o2::trigger::TF) {
151  if (mVerbose) {
152  LOG(INFO) << "Starting new TF for link FEEId 0x" << std::hex << std::setfill('0') << std::setw(6) << link.feeID;
153  }
154  if (mStartTFOnNewSPage && link.lastPageSize > 0) { // new TF being started, need to start new superpage
155  flushLinkSuperPage(link);
156  }
157  } else if (link.data.getSize() >= mSuperPageSize) {
158  flushLinkSuperPage(link);
159  }
160 
161  link.data.ensureFreeCapacity(MaxGBTPacketBytes + sizeof(RDH));
162  RDH rdh = mRDH; // new RDH to open
163  rdh.detectorField = mMAP.getRUDetectorField();
164  rdh.feeId = link.feeID;
165  rdh.linkID = link.id;
166  rdh.pageCnt = 0;
167  rdh.stop = 0;
168  rdh.packetCounter = link.packetCounter++;
169  link.lastRDH = reinterpret_cast<RDH*>(link.data.getEnd()); // pointer on last (to be written RDH for HBopen)
170 
171  link.data.addFast(reinterpret_cast<uint8_t*>(&rdh), sizeof(RDH)); // write RDH for new packet
172 }

188 template <class Mapping>
189 void MC2RawEncoder<Mapping>::addPageLinkHBF(GBTLink& link, bool stop)
190 {
192  if (!link.lastRDH) {
193  return;
194  }
195  // check if the superpage reached the size where it hase to be flushed
196  if (link.data.getSize() >= mSuperPageSize) {
197  flushLinkSuperPage(link);
198  }
199  auto rdh = link.lastRDH;
200  if (!rdh->stop) {
201  rdh->memorySize = link.data.getEnd() - reinterpret_cast<uint8_t*>(rdh); // set the size for the previous header RDH
202  rdh->offsetToNext = rdh->memorySize;
203  // pointer on last (to be written) RDH for HBclose
204  rdh = reinterpret_cast<RDH*>(link.data.getEnd()); // fetch pointer on to-be-written new RDH
205 
206  link.data.addFast(reinterpret_cast<uint8_t*>(link.lastRDH), sizeof(RDH)); // write RDH for old packet
207  if (mVerbose) {
208  LOG(INFO) << "Prev HBF for link FEEId 0x" << std::hex << std::setfill('0') << std::setw(6) << link.feeID;
209  if (mVerbose > 1) {
210  mHBFUtils.printRDH(*rdh);
211  }
212  }
213  rdh->offsetToNext = rdh->headerSize;
214  rdh->memorySize = rdh->headerSize;
215  rdh->pageCnt++;
216  rdh->stop = stop ? 0x1 : 0;
217  if (mVerbose) {
218  LOG(INFO) << (stop ? "Stop" : "Add") << " HBF for link FEEId 0x"
219  << std::hex << std::setfill('0') << std::setw(6) << link.feeID << " Pages: " << link.nTriggers;
220  if (mVerbose > 1 && stop) {
221  mHBFUtils.printRDH(*rdh);
222  }
223  }
224  link.lastRDH = rdh; // redirect to newly written RDH
225  }
226  link.lastPageSize = link.data.getSize(); // register current size of the superpage
227  link.nTriggers++; // number of pages on this superpage
228  if (stop) {
229  link.lastRDH = nullptr; // after closing it is not valid anymore
230  } else {
231  size_t offs = reinterpret_cast<uint8_t*>(link.lastRDH) - link.data.getPtr();
232  link.data.ensureFreeCapacity(MaxGBTPacketBytes + sizeof(RDH)); // make sure there is a room for next page
233  link.lastRDH = reinterpret_cast<RDH*>(&link.data[offs]); // fix link.lastRDH if the array was relocated
234  }
235  //
236 }

314 template <class Mapping>
315 void MC2RawEncoder<Mapping>::flushLinkSuperPage(GBTLink& link, FILE* outFl)
316 {
317  // write link superpage data to file
318  if (!outFl) {
319  outFl = mOutFile; // by default use global file
320  }
321  if (mVerbose) {
322  LOG(INFO) << "Flushing super page for link FEEId 0x" << std::hex << std::setfill('0') << std::setw(6)
323  << link.feeID << std::dec << std::setfill(' ') << " | size= " << link.lastPageSize << " bytes";
324  }
325  const auto ptr = link.data.getPtr(); // beginning of the buffer
326  fwrite(link.data.getPtr(), 1, link.lastPageSize, outFl); //write to file
327  link.data.setPtr(ptr + link.lastPageSize);
328  if (link.lastRDH) {
329  reinterpret_cast<uint8_t&>(link.lastRDH) -= link.lastPageSize; // move pointer on the last RDH
330  }
331  link.data.moveUnusedToHead(); // bring non-flushed data to top of the buffer
332  link.lastPageSize = 0; // will be updated on following closePageLinkHBF
333  link.nTriggers = 0;
334 }

337 template <class Mapping>
338 void MC2RawEncoder<Mapping>::fillGBTLinks(RUDecodeData& ru)
339 {
340  // fill data of the RU to links buffer, return the number of pages in the link with smallest amount of pages
341  constexpr uint8_t zero16[GBTPaddedWordLength] = {0}; // to speedup padding
342  ru.nCables = ru.ruInfo->nCables;
343 
344  // trigger word
345  GBTTrigger gbtTrigger;
346  gbtTrigger.bc = mCurrIR.bc;
347  gbtTrigger.orbit = mCurrIR.orbit;
348  gbtTrigger.internal = mIsContinuous;
349  gbtTrigger.triggerType = 0; //TODO
350 
351  for (int il = 0; il < MaxLinksPerRU; il++) {
352  auto link = ru.links[il].get();
353  if (!link) {
354  continue;
355  }
356  // number of words written (RDH included) on current page
357  int nGBTWords = int(link->data.getEnd() - reinterpret_cast<const uint8_t*>(link->lastRDH)) / GBTPaddedWordLength;
358  // estimate real payload size in GBT words
359  int nPayLoadWordsNeeded = 0;
360  for (int icab = ru.nCables; icab--;) { // calculate number of GBT words per link
361  if ((link->lanes & (0x1 << icab))) {
362  int nb = ru.cableData[icab].getSize();
363  nPayLoadWordsNeeded += nb ? 1 + (nb - 1) / 9 : 0; // single GBT word carries at most 9 payload bytes
364  }
365  }
366  if (nGBTWords >= MaxGBTWordsPerPacket - 3 - (nPayLoadWordsNeeded > 0)) {
367  // we write on same page if there is a room for trigger, header and trailer words + at least 1 payload word (if any)
368  addPageLinkHBF(*link); // open new page for the same trigger, repeating RDH of the same HBF
369  nGBTWords = int(link->data.getEnd() - reinterpret_cast<const uint8_t*>(link->lastRDH)) / GBTPaddedWordLength; // update counters
370  }
371  link->data.addFast(gbtTrigger.getW8(), GBTPaddedWordLength); // write GBT trigger
372  nGBTWords++;
373 
374  GBTDataHeader gbtHeader;
375  gbtHeader.packetIdx = 0;
376  gbtHeader.activeLanes = link->lanes;
377  link->data.addFast(gbtHeader.getW8(), GBTPaddedWordLength); // write GBT header
378  nGBTWords++;
379  //
380  GBTDataTrailer gbtTrailer; // lanes will be set on closing the trigger
381 
382  // now loop over the lanes served by this link, writing each time at most 9 bytes, untill all lanes are copied
383  do {
384  for (int icab = 0; icab < ru.nCables; icab++) {
385  if ((link->lanes & (0x1 << icab))) {
386  auto& cableData = ru.cableData[icab];
387  int nb = cableData.getUnusedSize();
388  if (!nb) {
389  continue; // write 80b word only if there is something to write
390  }
391  if (nb > 9) {
392  nb = 9;
393  }
394  int gbtWordStart = link->data.getSize(); // beginning of the current GBT word in the link
395  link->data.addFast(cableData.getPtr(), nb); // fill payload of cable
396  link->data.addFast(zero16, GBTPaddedWordLength - nb); // fill the rest of the GBT word by 0
397  link->data[gbtWordStart + 9] = mMAP.getGBTHeaderRUType(ru.ruInfo->ruType, ru.cableHWID[icab]); // set cable flag
398  cableData.setPtr(cableData.getPtr() + nb);
399  nPayLoadWordsNeeded--; // number of payload GBT words left
400  if (++nGBTWords == MaxGBTWordsPerPacket - 1) { // check if current GBT packet must be created (reserve 1 word for trailer)
401  if (nPayLoadWordsNeeded) { // we need to write the rest of the data on the new page
402  link->data.add(gbtTrailer.getW8(), GBTPaddedWordLength); // write empty GBT trailer for current packet
403  addPageLinkHBF(*link); // open new page for the same trigger, repeating RDH of the same HBF
404  link->data.addFast(gbtTrigger.getW8(), GBTPaddedWordLength); // repeate GBT trigger word
405  gbtHeader.packetIdx++;
406  link->data.addFast(gbtHeader.getW8(), GBTPaddedWordLength); // repeate GBT header
407  // update counters
408  nGBTWords = int(link->data.getEnd() - reinterpret_cast<const uint8_t*>(link->lastRDH)) / GBTPaddedWordLength;
409  }
410  }
411  } // storing data of single cable
412  } // loop over cables of this link
413  } while (nPayLoadWordsNeeded);
414  //
415  // all payload was dumped, write final trailer
416 
417  gbtTrailer.lanesStops = link->lanes;
418  gbtTrailer.packetDone = true;
419  link->data.addFast(gbtTrailer.getW8(), GBTPaddedWordLength); // write GBT trailer for the last packet
420 
421  } // loop over links of RU
422  ru.clearTrigger();
423  ru.nChipsFired = 0;
424 }
