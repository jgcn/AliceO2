35 void Digitizer::init()
36 {
37  mSimParam = SimParam::getInstance();
38  mRandomGenerator = new TRandom3(std::chrono::high_resolution_clock::now().time_since_epoch().count());
39 
40  float tau = mSimParam->getTimeResponseTau();
41  float N = mSimParam->getTimeResponsePower();
42  mTimeBinOffset = ((int)tau + 0.5);
43  mSignalFractionInTimeBins.clear();
44  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
45  RawResponse.SetParameters(1., 0., tau, N, 0.);
46  double RawResponseTotalIntegral = (N > 0) ? pow(N, -N) * exp(N) * std::tgamma(N) : 1.;
47  for (int j = 0; j < constants::EMCAL_MAXTIMEBINS; j++) {
48  double val = RawResponse.Integral(-mTimeBinOffset - 0.5 + j, -mTimeBinOffset + 0.5 + j) / RawResponseTotalIntegral;
49  if (val < 1e-10) {
50  break;
51  }
52  mSignalFractionInTimeBins.push_back(val);
53  }
54 }

57 double Digitizer::rawResponseFunction(double* x, double* par)
58 {
59  double signal = 0.;
60  double tau = par[2];
61  double n = par[3];
62  double ped = par[4];
63  double xx = (x[0] - par[1] + tau) / tau;
64 
65  if (xx <= 0) {
66  signal = ped;
67  } else {
68  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
69  }
70 
71  return signal;
72 }

78 void Digitizer::process(const std::vector<Hit>& hits, std::vector<Digit>& digits)
79 {
80  digits.clear();
81  mDigits.clear();
82  mMCTruthContainer.clear();
83 
84  if (mSimulateNoiseDigits) {
85  addNoiseDigits();
86  }
87 
88  for (auto hit : hits) {
89  try {
90  hitToDigits(hit);
91 
92  for (auto digit : mTempDigitVector) {
93  Int_t id = digit.getTower();
94 
95  if (id < 0 || id > mGeometry->GetNCells()) {
96  LOG(WARNING) << "tower index out of range: " << id;
97  continue;
98  }
99 
100  MCLabel label(hit.GetTrackID(), mCurrEvID, mCurrSrcID, false, 1.0);
101  LabeledDigit d(digit, label);
102  mDigits[id].push_back(d);
103  }
104  } catch (InvalidPositionException& e) {
105  LOG(ERROR) << "Error in creating the digit: " << e.what();
106  }
107  }
108 
109  fillOutputContainer(digits);
110 }

113 void Digitizer::hitToDigits(const Hit& hit)
114 {
115  mTempDigitVector.clear();
116  Int_t tower = hit.GetDetectorID();
117  Double_t energy = hit.GetEnergyLoss();
118 
119  if (mSimulateTimeResponse) {
120  for (int j = 0; j < mSignalFractionInTimeBins.size(); j++) {
121  double val = energy * mSignalFractionInTimeBins.at(j);
122  if ((val < mSimParam->getTimeResponseThreshold()) && (j > mTimeBinOffset)) {
123  break;
124  }
125  Digit digit(tower, val, mEventTime + (j - mTimeBinOffset) * constants::EMCAL_TIMESAMPLE);
126  mTempDigitVector.push_back(digit);
127  }
128  } else {
129  Digit digit(tower, energy, mEventTime);
130  mTempDigitVector.push_back(digit);
131  }
132 }

161 void Digitizer::fillOutputContainer(std::vector<Digit>& digits)
162 {
163  std::list<LabeledDigit> l;
164 
165  for (auto t : mDigits) {
166  std::list<LabeledDigit> tower = t.second;
167  tower.sort();
168 
169  while (!tower.empty()) {
170  LabeledDigit ld1 = tower.front();
171  tower.pop_front();
172 
173  // loop over all other entries in the container, check if we can add the digits
174  std::vector<decltype(tower.begin())> toDelete;
175  for (auto ld2 = tower.begin(); ld2 != tower.end(); ++ld2) { // must be iterator in order to know the position in the container for erasing
176  if (ld1.canAdd(*ld2)) {
177  ld1 += *ld2;
178  toDelete.push_back(ld2);
179  }
180  }
181  for (auto del : toDelete) {
182  tower.erase(del);
183  }
184 
185  if (mSmearEnergy) {
186  smearEnergy(ld1);
187  }
188 
189  if (!mRemoveDigitsBelowThreshold || (ld1.getEnergy() >= mSimParam->getDigitThreshold() * (constants::EMCAL_ADCENERGY))) {
190  l.push_back(ld1);
191  }
192  }
193  }
194 
195  l.sort();
196 
197  for (auto d : l) {
198  Digit digit = d.getDigit();
199  std::vector<MCLabel> labels = d.getLabels();
200  digits.push_back(digit);
201 
202  Int_t LabelIndex = mMCTruthContainer.getIndexedSize();
203  for (auto label : labels) {
204  mMCTruthContainer.addElementRandomAccess(LabelIndex, label);
205  }
206  }
207 }

210 void Digitizer::smearEnergy(LabeledDigit& digit)
211 {
212  Double_t energy = digit.getEnergy();
213  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
214  energy *= mRandomGenerator->Poisson(fluct) / fluct;
215  digit.setEnergy(energy);
216 }
