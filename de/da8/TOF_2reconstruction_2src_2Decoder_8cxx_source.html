25 bool Decoder::open(std::string name)
26 {
27  if (mFile.is_open()) {
28  std::cout << "Warning: a file was already open, closing" << std::endl;
29  mFile.close();
30  }
31  mFile.open(name.c_str(), std::fstream::in | std::fstream::binary);
32  if (!mFile.is_open()) {
33  std::cerr << "Cannot open " << name << std::endl;

37  mFile.seekg(0, mFile.end);

40 
41  mBufferLocal.resize(mSize);

44  // read content of infile
45  mFile.read(mBuffer, mSize);
46  mUnion = reinterpret_cast<Union_t*>(mBuffer);
47  mUnionEnd = reinterpret_cast<Union_t*>(mBuffer + mSize - 1);

50  return false;

60 void Decoder::readTRM(std::vector<Digit>* digits, int iddl, int orbit, int bunchid)
61 {
62  if (mVerbose)

64  int nhits = mUnion->frameHeader.numberOfHits;

88 void Decoder::fromRawHit2Digit(int iddl, int itrm, int itdc, int ichain, int channel, int orbit, int bunchid, int tdc, int tot, std::array<int, 4>& digitInfo)
89 {
90  // convert raw info in digit info (channel, tdc, tot, bc)
91  // tdc = packetHit.time + (frameHeader.frameID << 13)
92  int echannel = Geo::getECHFromIndexes(iddl, itrm, ichain, itdc, channel);
93  digitInfo[0] = Geo::getCHFromECH(echannel);
94  digitInfo[2] = tot;
95 
96  digitInfo[3] = int(orbit * o2::tof::Geo::BC_IN_ORBIT);
97  digitInfo[3] += bunchid;
98  digitInfo[3] += tdc / 1024;
99  digitInfo[1] = tdc % 1024;
100 }

102 bool Decoder::decode(std::vector<Digit>* digits) // return a vector of digits in a TOF readout window
103 {
104 #ifdef VERBOSE
105  if (mVerbose)
106  std::cout << "-------- START DECODE EVENT ----------------------------------------" << std::endl;
107 #endif
108  auto start = std::chrono::high_resolution_clock::now();
109 
110  if (mUnion > mUnionEnd)
111  return 1;
112 
113  // .. decoding
114 
115  // int eventcounter;
116  for (int icrate = 0; icrate < 72; icrate++) {
117  // read Crate Header
118  //eventcounter = mUnion->crateHeader.eventCounter;
119  int bunchid = mUnion->crateHeader.bunchID;
120  if (mVerbose)
121  printCrateInfo();
122  mUnion++;
123 
124  //read Orbit
125  int orbit = mUnion->crateOrbit.orbitID;
126  if (mVerbose)
127  printf("orbit ID = %d\n", orbit);
128  mUnion++;
129 
130  while (!mUnion->frameHeader.mustBeZero) {
131  readTRM(digits, icrate, orbit, bunchid);
132  }
133 
134  // read Crate Tralier
135  if (mVerbose)
136  printCrateTrailerInfo();
137  mUnion++;
138  }
139 
140  auto finish = std::chrono::high_resolution_clock::now();
141  std::chrono::duration<double> elapsed = finish - start;
142  mIntegratedTime = elapsed.count();
143 
144 #ifdef VERBOSE
145  if (mVerbose && mIntegratedTime)
146  std::cout << "-------- END DECODE EVENT ------------------------------------------"
147  << " " << mIntegratedBytes << " words"
148  << " | " << 1.e3 * mIntegratedTime << " ms"
149  << " | " << 1.e-6 * mIntegratedBytes / mIntegratedTime << " MB/s (average)"
150  << std::endl;
151 #endif
152 
153  return 0;
154 }
