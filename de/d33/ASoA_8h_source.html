90 template <typename T, typename ChunkingPolicy = Chunked>
91 class ColumnIterator : ChunkingPolicy
92 {
93  public:
98  ColumnIterator(arrow::Column const* column)
99  : mColumn{column},
100  mCurrentPos{nullptr},
101  mFirstIndex{0},
102  mCurrentChunk{0}
103  {
104  auto chunks = mColumn->data();
105  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
106  mCurrent = array->raw_values();
107  mLast = mCurrent + array->length();
108  }
109 
110  ColumnIterator() = default;
111  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
112  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
113 
114  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
115  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
116 
118  void nextChunk() const
119  {
120  auto chunks = mColumn->data();
121  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
122  mFirstIndex += previousArray->length();
123  mCurrentChunk++;
124  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
125  mCurrent = array->raw_values() - mFirstIndex;
126  mLast = mCurrent + array->length() + mFirstIndex;
127  }
128 
129  void prevChunk() const
130  {
131  auto chunks = mColumn->data();
132  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
133  mFirstIndex -= previousArray->length();
134  mCurrentChunk--;
135  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
136  mCurrent = array->raw_values() - mFirstIndex;
137  mLast = mCurrent + array->length() + mFirstIndex;
138  }
139 
140  void moveToChunk(int chunk)
141  {
142  if (mCurrentChunk < chunk) {
143  while (mCurrentChunk != chunk) {
144  nextChunk();
145  }
146  } else {
147  while (mCurrentChunk != chunk) {
148  prevChunk();
149  }
150  }
151  }
152 
154  void moveToEnd()
155  {
156  mCurrentChunk = mColumn->data()->num_chunks() - 1;
157  auto chunks = mColumn->data();
158  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
159  assert(array.get());
160  mFirstIndex = mColumn->length() - array->length();
161  mCurrent = array->raw_values() - mFirstIndex;
162  mLast = mCurrent + array->length() + mFirstIndex;
163  }
164 
165  T const& operator*() const
166  {
167  if constexpr (ChunkingPolicy::chunked) {
168  if (O2_BUILTIN_UNLIKELY(((mCurrent + *mCurrentPos) >= mLast))) {
169  nextChunk();
170  }
171  }
172  return *(mCurrent + *mCurrentPos);
173  }
174 
175  // Move to the chunk which containts element pos
176  ColumnIterator<T>& moveToPos()
177  {
178  // If we get outside range of the current chunk, go to the next.
179  if constexpr (ChunkingPolicy::chunked) {
180  while (O2_BUILTIN_UNLIKELY((mCurrent + *mCurrentPos) >= mLast)) {
181  nextChunk();
182  }
183  }
184  return *this;
185  }
186 
187  // Move to the chunk which containts element pos
188  ColumnIterator<T>& checkNextChunk()
189  {
190  if constexpr (ChunkingPolicy::chunked) {
191  if (O2_BUILTIN_LIKELY((mCurrent + *mCurrentPos) <= mLast)) {
192  return *this;
193  }
194  nextChunk();
195  }
196  return *this;
197  }
198 
199  mutable T const* mCurrent;
200  int64_t const* mCurrentPos;
201  mutable T const* mLast;
202  arrow::Column const* mColumn;
203  mutable int mFirstIndex;
204  mutable int mCurrentChunk;
205 };
