265 bool Encoder::encode(std::vector<Digit> summary, int tofwindow) // pass a vector of digits in a TOF readout window, tof window is the entry in the vector-of-vector of digits needed to extract bunch id and orbit
266 {
267 
268  mEventCounter = tofwindow; // tof window index
269  mOrbitID = mEventCounter / Geo::NWINDOW_IN_ORBIT; // since 3 tof window = 1 orbit
270  mBunchID = ((mEventCounter % Geo::NWINDOW_IN_ORBIT) * Geo::BC_IN_ORBIT) / Geo::NWINDOW_IN_ORBIT; // bunch crossing in the current orbit at the beginning of the window
271 
272  if (!summary.size()) {
273  for (int iemptycrate = 0; iemptycrate < 72; iemptycrate++) {
274  encodeEmptyCrate(iemptycrate);
275  }
276  return 1; // empty array
277  }
278 
279  // caching electronic indexes in digit array
280  for (auto dig = summary.begin(); dig != summary.end(); dig++) {
281  int digitchannel = dig->getChannel();
282  dig->setElectronicIndex(Geo::getECHFromCH(digitchannel));
283  }
284 
285  // sorting by electroni indexes
286  std::sort(summary.begin(), summary.end(),
287  [](Digit a, Digit b) { return a.getElectronicIndex() < b.getElectronicIndex(); });
288 
289 #ifdef VERBOSE
290  if (mVerbose)
291  std::cout << "-------- START ENCODE EVENT ----------------------------------------" << std::endl;
292 #endif
293  auto start = std::chrono::high_resolution_clock::now();
294 
295  int currentEvent = 0;
296  int currentCrate = summary[0].getElCrateIndex();
297 
298  for (int iemptycrate = 0; iemptycrate < currentCrate; iemptycrate++) {
299  encodeEmptyCrate(iemptycrate);
300  }
301 
302  while (currentCrate > -1) { // process also empty crates
303  int prevCrate = currentCrate;
304  currentCrate = encodeCrate(summary, currentCrate, currentEvent);
305 
306  if (currentCrate == -1) {
307  for (int iemptycrate = prevCrate + 1; iemptycrate < 72; iemptycrate++) {
308  encodeEmptyCrate(iemptycrate);
309  }
310  } else {
311  for (int iemptycrate = prevCrate + 1; iemptycrate < currentCrate; iemptycrate++) {
312  encodeEmptyCrate(iemptycrate);
313  }
314  }
315  }
316 
317  auto finish = std::chrono::high_resolution_clock::now();
318  std::chrono::duration<double> elapsed = finish - start;
319  mIntegratedTime = elapsed.count();
320 
321 #ifdef VERBOSE
322  if (mVerbose && mIntegratedTime)
323  std::cout << "-------- END ENCODE EVENT ------------------------------------------"
324  << " " << mIntegratedBytes << " words"
325  << " | " << 1.e3 * mIntegratedTime << " ms"
326  << " | " << 1.e-6 * mIntegratedBytes / mIntegratedTime << " MB/s (average)"
327  << std::endl;
328 #endif
329 
330  return false;
331 }
