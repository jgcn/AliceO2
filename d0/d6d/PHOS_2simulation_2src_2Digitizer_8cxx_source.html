33 void Digitizer::process(const std::vector<Hit>& hits, std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::phos::MCLabel>& labels)
34 {
35  // Convert list of hits to digits:
36  // Add hits with energy deposition in same cell and same time
37  // Add energy corrections
38  // Apply time smearing
39 
40  Int_t hitIndex = 0;
41  Int_t hitAbsId = 0;
42  Int_t nHits = hits.size();
43  Hit hit;
44  if (hitIndex < nHits) {
45  hit = hits.at(hitIndex);
46  hitAbsId = hit.GetDetectorID();
47  }
48 
49  Int_t nTotCells = mGeometry->GetTotalNCells();
50  for (Int_t absId = 1; absId < nTotCells; absId++) {
51 
52  // If signal exist in this cell, add noise to it, otherwise just create noise digit
53  if (absId == hitAbsId) {
54  int labelIndex = labels.getIndexedSize();
55  //Add primary info: create new MCLabels entry
56  o2::phos::MCLabel label(hit.GetTrackID(), mCurrEvID, mCurrSrcID, true, hit.GetEnergyLoss());
57  labels.addElement(labelIndex, label);
58 
59  Digit digit(hit, labelIndex);
60 
61  hitIndex++;
62  if (hitIndex < nHits) {
63  Hit hitNext = hits.at(hitIndex);
64  Digit digitNext(hitNext, -1); //Do not create MCTruth entry so far
65  while ((hitIndex < nHits) && digit.canAdd(digitNext)) {
66  digit += digitNext;
67 
68  //add MCLabel to list (add energy if same primary or add another label)
69  o2::phos::MCLabel label(hitNext.GetTrackID(), mCurrEvID, mCurrSrcID, true, hitNext.GetEnergyLoss());
70  labels.addElementRandomAccess(labelIndex, label);
71 
72  hitIndex++;
73  if (hitIndex < nHits) {
74  hitNext = hits.at(hitIndex);
75  digitNext.FillFromHit(hitNext);
76  }
77  }
78  if (hitIndex < nHits) {
79  hitAbsId = digitNext.getAbsId();
80  } else {
81  hitAbsId = 0;
82  }
83  hit = hits.at(hitIndex);
84  } else {
85  hitAbsId = 99999; // out of PHOS
86  }
87  //Current digit finished, sort MCLabels according to eDeposited
88  auto lbls = labels.getLabels(labelIndex);
89  std::sort(lbls.begin(), lbls.end(),
90  [](o2::phos::MCLabel a, o2::phos::MCLabel b) { return a.getEdep() > b.getEdep(); });
91 
92  // Add Electroinc noise, apply non-linearity, digitize, de-calibrate, time resolution
93  Double_t energy = digit.getAmplitude();
94  // Simulate electronic noise
95  energy += SimulateNoiseEnergy();
96 
97  if (o2::phos::PHOSSimParams::Instance().mApplyNonLinearity) {
98  energy = NonLinearity(energy);
99  }
100  if (o2::phos::PHOSSimParams::Instance().mApplyDigitization) {
101  energy = DigitizeEnergy(energy);
102  }
103  if (o2::phos::PHOSSimParams::Instance().mApplyDecalibration) {
104  energy = Decalibrate(energy);
105  }
106  digit.setAmplitude(energy);
107  if (o2::phos::PHOSSimParams::Instance().mApplyTimeResolution) {
108  digit.setTimeStamp(TimeResolution(digit.getTimeStamp(), energy));
109  }
110  digits.push_back(digit);
111  } else { // No signal in this cell,
112  if (!mGeometry->IsCellExists(absId)) {
113  continue;
114  }
115  // Simulate noise
116  Double_t energy = SimulateNoiseEnergy();
117  Double_t time = SimulateNoiseTime();
118  if (energy > o2::phos::PHOSSimParams::Instance().mZSthreshold) {
119  Digit noiseDigit(absId, energy, time, -1); // current AbsId, energy, random time, no primary
120  digits.push_back(noiseDigit);
121  }
122  }
123  }
124 }

127 Double_t Digitizer::NonLinearity(const Double_t e)
128 {
129  double a = o2::phos::PHOSSimParams::Instance().mCellNonLineaityA;
130  double b = o2::phos::PHOSSimParams::Instance().mCellNonLineaityB;
131  double c = o2::phos::PHOSSimParams::Instance().mCellNonLineaityC;
132  return e * c * (1. + a * exp(-e * e / 2. / b / b));
133 }

148 Double_t Digitizer::TimeResolution(const Double_t time, const Double_t e)
149 {
150  // apply time resolution
151 
152  Double_t timeResolution = o2::phos::PHOSSimParams::Instance().mTimeResolutionA +
153  o2::phos::PHOSSimParams::Instance().mTimeResolutionB /
154  std::max(float(e), o2::phos::PHOSSimParams::Instance().mTimeResThreshold);
155  return gRandom->Gaus(time, timeResolution);
156 }
