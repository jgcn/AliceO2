38 template <typename T>
39 constexpr auto selectArrowType()
40 {
41  if constexpr (std::is_same_v<T, int>) {
42  return atype::INT32;
43  } else if constexpr (std::is_same_v<T, bool>) {
44  return atype::BOOL;
45  } else if constexpr (std::is_same_v<T, float>) {
46  return atype::FLOAT;
47  } else if constexpr (std::is_same_v<T, double>) {
48  return atype::DOUBLE;
49  } else {
50  return atype::NA;
51  }
52  O2_BUILTIN_UNREACHABLE();
53 }
