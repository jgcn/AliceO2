48 inline int Segmentation::findPadByFEE(int dualSampaId, int dualSampaChannel) const
49 {
50  bool isBending = dualSampaId < 1024;
51  int catPadIndex;
52  if (isBending) {
53  catPadIndex = mBending.findPadByFEE(dualSampaId, dualSampaChannel);
54  if (!mBending.isValid(catPadIndex)) {
55  return -1;
56  }
57  } else {
58  catPadIndex = mNonBending.findPadByFEE(dualSampaId, dualSampaChannel);
59  if (!mNonBending.isValid(catPadIndex)) {
60  return -1;
61  }
62  }
63  return padC2DE(catPadIndex, isBending);
64 }

82 inline void Segmentation::catSegPad(int dePadIndex, const CathodeSegmentation*& catseg, int& padcuid) const
83 {
84  if (!isValid(dePadIndex)) {
85  catseg = nullptr;
86  return;
87  }
88  if (dePadIndex < mPadIndexOffset) {
89  catseg = &mBending;

92  }
93  catseg = &mNonBending;
94  padcuid = dePadIndex - mPadIndexOffset;

97 inline bool Segmentation::findPadPairByPosition(double x, double y, int& b, int& nb) const
98 {
99  b = mBending.findPadByPosition(x, y);
100  nb = mNonBending.findPadByPosition(x, y);

102  return false;
103  }
104  b = padC2DE(b, true);
105  nb = padC2DE(nb, false);
106  return true;

145 inline std::string Segmentation::padAsString(int dePadIndex) const
146 {
147  if (!isValid(dePadIndex)) {

149  }
150  const CathodeSegmentation* catSeg{ nullptr };
