112  {
113  if (mNTracks >= mNMaxTracks) {
114 #ifndef GPUCA_GPUCODE
115  GPUError("Error: Track dropped (no memory available) -> must not happen");
116 #endif
117  return (1);
118  }
119 #ifdef GPUCA_ALIROOT_LIB
120  new (&mTracks[mNTracks++]) GPUTRDTrack(trk); // We need placement new, since the class is virtual
121 #else
122  mTracks[mNTracks++] = trk;
123 #endif
124  if (label >= 0) {
125  mTracks[mNTracks - 1].SetLabel(label);
126  }
127  if (nTrkltsOffline) {
128  for (int i = 0; i < 4; ++i) {
129  mTracks[mNTracks - 1].SetNtrackletsOffline(i, nTrkltsOffline[i]); // see GPUTRDTrack.h for information on the index
130  }
131  }
132  mTracks[mNTracks - 1].SetLabelOffline(labelOffline);
133  return (0);
134  }
