79 template <class T>
80 GPUdi() T* GPUCommonAlgorithm::QuicksortPartition(T* left, T* right)
81 {
82  T* mid = left + ((right - left) / 2);
83  T pivot = *mid;
84  SortSwap(mid, left);
85  T* i = left + 1;
86  T* j = right;
87  while (i <= j) {
88  while (i <= j && *i <= pivot) {
89  i++;
90  }
91  while (i <= j && *j > pivot) {
92  j--;
93  }
94  if (i < j) {
95  SortSwap(i, j);
96  }
97  }
98  SortSwap(i - 1, left);
99  return i - 1;
100 }

102 template <class T, class S>
103 GPUdi() T* GPUCommonAlgorithm::QuicksortPartition(T* left, T* right, const S& comp)
104 {
105  T* mid = left + ((right - left) / 2);
106  SortSwap(mid, right);
107  T* pivot = right;
108  T* i = left;
109  T* j = right - 1;
110  while (i <= j) {
111  while (i <= j && !comp(*j, *pivot)) {
112  j--;
113  }
114  while (i <= j && comp(*i, *pivot)) {
115  i++;
116  }
117  if (i < j) {
118  SortSwap(i, j);
119  }
120  }
121  SortSwap(j + 1, right);
122  return j + 1;
123 }

125 template <class T>
126 GPUdi() void GPUCommonAlgorithm::Quicksort(T* left, T* right)
127 {
128  if (left >= right) {
129  return;
130  }
131  if (right - left <= 4) {
132  Insertionsort(left, right);
133  return;
134  }
135  T* part = QuicksortPartition(left, right);
136 
137  Quicksort(left, part - 1);
138  Quicksort(part + 1, right);
139 }

157 template <class T>
158 GPUdi() void GPUCommonAlgorithm::Insertionsort(T* left, T* right)
159 {
160  if (left >= right) {
161  return;
162  }
163  while (left < right) {
164  T* min = left;
165  for (T* test = left + 1; test <= right; test++) {
166  if (*test < *min) {
167  min = test;
168  }
169  }
170  if (min != left) {
171  SortSwap(left, min);
172  }
173  left++;
174  }
175 }

178 GPUdi() void GPUCommonAlgorithm::Insertionsort(T* left, T* right, const S& comp)
179 {

181  return;
182  }

184  T* min = left;

187  min = test;
188  }
189  }

191  SortSwap(left, min);
192  }
193  left++;
194  }
