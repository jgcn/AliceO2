47  void run(framework::ProcessingContext& pc)
48  {
49  static bool finished = false;
50  if (finished) {
51  return;
52  }
53  // get digit data
54  auto digits = pc.inputs().get<std::vector<std::vector<o2::tof::Digit>>*>("tofdigits");
55  auto labelvector = std::make_shared<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>>();
56  if (mUseMC) {
57  auto digitlabels = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
58  *labelvector.get() = std::move(*digitlabels);
59  mClusterer.setMCTruthContainer(&mClsLabels);
60  mClsLabels.clear();
61  }
62  // call actual clustering routine
63  mClustersArray.clear();
64 
65  for (int i = 0; i < digits->size(); i++) {
66  printf("# TOF readout window for clusterization = %i\n", i);
67  auto digitsRO = digits->at(i);
68  mReader.setDigitArray(&digitsRO);
69  if (mUseMC) {
70  mClusterer.process(mReader, mClustersArray, &(labelvector->at(i)));
71  } else
72  mClusterer.process(mReader, mClustersArray, nullptr);
73  }
74  LOG(INFO) << "TOF CLUSTERER : TRANSFORMED " << digits->size()
75  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
76 
77  // send clusters
78  pc.outputs().snapshot(Output{"TOF", "CLUSTERS", 0, Lifetime::Timeframe}, mClustersArray);
79  // send labels
80  if (mUseMC)
81  pc.outputs().snapshot(Output{"TOF", "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mClsLabels);
82 
83  // declare done
84  finished = true;
85  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
86  }
87 
88  private:
89  DigitDataReader mReader;
90  Clusterer mClusterer;
91 
92  std::vector<Cluster> mClustersArray;
93  MCLabelContainer mClsLabels;
94 };
