148 void render3D()
149 {
150  int cur_width, cur_height;
151  glfwGetFramebufferSize(g3dCtx.window, &cur_width, &cur_height);
152 
153  int lastDrawState = -1;
154  int lastUniform = -1;
155 
156  // Cleaning happens somewhere else
157  sg_begin_default_pass(&g3dCtx.passes.back(), cur_width, cur_height);
158 
159  for (auto& command : g3dCtx.commands) {
161  if (command.drawStateIdx != lastDrawState) {
162  sg_draw_state const& state = g3dCtx.drawStates[command.drawStateIdx];
163  lastDrawState = command.drawStateIdx;
164  sg_apply_draw_state(&state);
165  }
167  if (command.uniformIdx != lastUniform) {
168  vs_params_t const& params = g3dCtx.params[command.uniformIdx];
169  sg_apply_uniform_block(SG_SHADERSTAGE_VS, 0, &params, sizeof(params));
170  lastUniform = command.uniformIdx;
171  }
173  sg_draw(command.start, command.end, command.instances);
174  }
175  sg_end_pass();
176  sg_commit();
177 }
