98  void getChipInfoSW(int chipSW, ChipInfo& chInfo) const
99  {
100  chInfo.id = chipSW;
101  if (chipSW > NChipsSB[IB] + NChipsSB[MB] - 1) { // OB
102  chipSW -= NChipsSB[IB] + NChipsSB[MB];
103  chInfo.ruType = OB;
104  auto dvRU = std::div(chipSW, NChipsPerStaveSB[OB]);
105  chInfo.ru = NStavesSB[IB] + NStavesSB[MB] + dvRU.quot; // RU ID == stave ID
106  chInfo.chOnRU = getChipOnRUInfo(OB, dvRU.rem);
107  } else if (chipSW > NChipsSB[IB] - 1) { // MB
108  chipSW -= NChipsSB[IB];
109  chInfo.ruType = MB;
110  auto dvRU = std::div(chipSW, NChipsPerStaveSB[MB]);
111  chInfo.ru = NStavesSB[IB] + dvRU.quot; // RU ID == stave ID
112  chInfo.chOnRU = getChipOnRUInfo(MB, dvRU.rem);
113  } else { // IB
114  chInfo.ruType = IB;
115  auto dvRU = std::div(chipSW, NChipsPerStaveSB[IB]);
116  chInfo.ru = dvRU.quot; // RU ID == stave ID = module ID
117  chInfo.chOnRU = getChipOnRUInfo(IB, dvRU.rem);
118  }
119  }
