62 // customize clusterers and cluster decoders to process immediately what comes in
63 void customize(std::vector<o2::framework::CompletionPolicy>& policies)
64 {
65  // we customize the processors to consume data as it comes
66  policies.push_back({"tpc-sector-processors",
67  [](o2::framework::DeviceSpec const& spec) {
68  // the decoder should process immediately
69  bool apply = spec.name.find("decoder") != std::string::npos || spec.name.find("clusterer") != std::string::npos;
70  if (apply) {
71  LOG(INFO) << "Applying completion policy 'consume' to device " << spec.name;
72  }
73  return apply;
74  },
75  [](gsl::span<o2::framework::PartRef const> const& inputs) {
76  o2::framework::CompletionPolicy::CompletionOp policy = o2::framework::CompletionPolicy::CompletionOp::Wait;
77  int nValidParts = 0;
78  bool eod = false;
79  if (!gDoMC) {
80  for (auto const& part : inputs) {
81  if (part.header == nullptr) {
82  continue;
83  }
84  nValidParts++;
85  auto const* header = o2::header::get<o2::header::DataHeader*>(part.header->GetData(), part.header->GetSize());
86  auto const* sectorHeader = o2::header::get<o2::tpc::TPCSectorHeader*>(part.header->GetData(), part.header->GetSize());
87  if (sectorHeader && sectorHeader->sector < 0) {
88  eod = true;
89  }
90  }
91  if (nValidParts == inputs.size()) {
92  policy = o2::framework::CompletionPolicy::CompletionOp::Consume;
93  } else if (eod == false) {
94  policy = o2::framework::CompletionPolicy::CompletionOp::Consume;
95  }
96  return policy;
97  }
98  using IndexType = o2::header::DataHeader::SubSpecificationType;
99 
100  std::set<IndexType> matchIndices;
101  for (auto const& part : inputs) {
102  if (part.header == nullptr) {
103  continue;
104  }
105  nValidParts++;
106  auto const* header = o2::header::get<o2::header::DataHeader*>(part.header->GetData(), part.header->GetSize());
107  assert(header != nullptr);
108  auto haveAlready = matchIndices.find(header->subSpecification);
109  if (haveAlready != matchIndices.end()) {
110  // inputs should be data-mc pairs if the index is already in the list
111  // the pair is complete and we can remove the index
112  matchIndices.erase(haveAlready);
113  } else {
114  // store the index in order to check if there is a pair
115  matchIndices.emplace(header->subSpecification);
116  }
117  auto const* sectorHeader = o2::header::get<o2::tpc::TPCSectorHeader*>(part.header->GetData(), part.header->GetSize());
118  if (sectorHeader && sectorHeader->sector < 0) {
119  eod = true;
120  }
121  }
122  if (nValidParts == inputs.size()) {
123  policy = o2::framework::CompletionPolicy::CompletionOp::Consume;
124  } else if (matchIndices.size() == 0 && eod == false) {
125  policy = o2::framework::CompletionPolicy::CompletionOp::Consume;
126  }
127  return policy;
128  }});
129 }

149 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
150 {
151  auto tpcSectors = o2::RangeTokenizer::tokenize<int>(cfgc.options().get<std::string>("tpc-sectors"));
152  // the lane configuration defines the subspecification ids to be distributed among the lanes.
153  std::vector<int> laneConfiguration;
154  auto nLanes = cfgc.options().get<int>("tpc-lanes");
155  auto inputType = cfgc.options().get<std::string>("input-type");
156  if (inputType == "digitizer") {
157  // the digitizer is using a different lane setup so we have to force this for the moment
158  laneConfiguration.resize(nLanes);
159  std::iota(laneConfiguration.begin(), laneConfiguration.end(), 0);
160  } else {
161  laneConfiguration = tpcSectors;
162  }
163 
164  // set up configuration
165  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
166  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
167  o2::conf::ConfigurableParam::writeINI("o2tpcrecoworkflow_configuration.ini");
168 
169  gDoMC = not cfgc.options().get<bool>("disable-mc");
170  auto dispmode = cfgc.options().get<std::string>("dispatching-mode");
171  gDispatchPrompt = !(dispmode == "single");
172  return o2::tpc::reco_workflow::getWorkflow(tpcSectors, // sector configuration
173  laneConfiguration, // lane configuration
174  gDoMC, //
175  nLanes, //
176  inputType, //
177  cfgc.options().get<std::string>("output-type"), //
178  cfgc.options().get<bool>("ca-clusterer") //
179  );
