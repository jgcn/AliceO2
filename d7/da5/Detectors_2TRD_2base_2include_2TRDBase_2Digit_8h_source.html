65  static void convertMapToVectors(const SignalContainer_t& adcMapCont,
66  DigitContainer_t& digitCont)
67  {
68  //
69  // Create a digit and a digit-index container from a map container
70  //
71  for (const auto& element : adcMapCont) {
72  const int key = element.first;
73  ArrayADC_t adcs{};
74  for (int i = 0; i < kTB; ++i) {
75  adcs[i] = element.second[i];
76  }
77  size_t idx = element.second[kTB];
78  digitCont.emplace_back(Digit::getDetectorFromKey(key),
79  Digit::getRowFromKey(key),
80  Digit::getColFromKey(key),
81  adcs, idx);
82  }
83  }
84  static void convertVectorsToMap(const DigitContainer_t& digitCont,
85  SignalContainer_t& adcMapCont)
86  {
87  //
88  // Create a map container from a digit and a digit-index container
89  //
90  for (const auto& element : digitCont) {

92  element.getRow(),
93  element.getPad());
94  ArrayADCext_t adcsext{};
95  for (int i = 0; i < kTB; ++i) {
96  adcsext[i] = element.getADC()[i];
97  }
98  adcMapCont[key] = adcsext;
99  }
100  }
