92  bool readTF()
93  {
94  mBytes.clear();
95  while (readBlock()) {
96  ++mHBCounter;
97  if (mHBCounter % mHBperTimeframe == 0) {
98  return true;
99  }
100  }
101  return false;
102  }
103 
104  bool readBlock()
105  {
106  bool stop = false;
107  while (!stop) {
108  // Read header
109  read(raw::sHeaderSizeInBytes);
110 
111  // The check on the eof needs to be placed here and not at the beginning of the function.
112  // The reason is that the eof flag is set if we try to read after the eof
113  // But, since we know the size, we read up to the last character.
114  // So we turn on the eof flag only if we try to read past the last data.
115  // Of course, we resized the mBytes before trying to read.
116  // Since we read 0, we need to remove the last bytes
117  if (mFile.eof()) {
118  mBytes.resize(mBytes.size() - raw::sHeaderSizeInElements);
119  mState = 1;
120  return false;
121  }
122  const header::RAWDataHeader* rdh = reinterpret_cast<const header::RAWDataHeader*>(&(mBytes[mBytes.size() - raw::sHeaderSizeInElements]));
123  stop = rdh->stop;
124  if (rdh->offsetToNext > raw::sHeaderSizeInBytes) {
125  read(rdh->offsetToNext - raw::sHeaderSizeInBytes);
126  }
127  }
128 
129  return true;
130  }
