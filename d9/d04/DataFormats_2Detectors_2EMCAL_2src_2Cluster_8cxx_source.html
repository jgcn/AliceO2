41 Vector3D<Float_t> Cluster::getMomentum(const Point3D<Float_t>* vertex) const
42 {
43  std::array<Float_t, 3> pos = {{this->X(), this->Y(), this->Z()}};
44 
45  if (vertex) { //calculate direction from vertex
46  pos[0] -= vertex->X();
47  pos[1] -= vertex->Y();
48  pos[2] -= vertex->Z();
49  }
50 
51  Float_t r = sqrt(pos[0] * pos[0] +
52  pos[1] * pos[1] +
53  pos[2] * pos[2]);
54 
55  if (std::abs(r) < DBL_EPSILON)
56  throw InvalidPositionException(pos);
57  return {mEnergy * pos[0] / r, mEnergy * pos[1] / r, mEnergy * pos[2] / r};
58 }
