91 Bool_t GeneratorHepMC::addTracks(FairPrimaryGenerator* primGen) const
92 {
96  auto particles = mEvent->particles();
97  for (auto const& particle : particles) {
98 
100  auto pdg = particle->pid();
101  auto st = particle->status();
102  auto momentum = particle->momentum();
103  auto vertex = particle->production_vertex()->position();
104  auto parents = particle->parents(); // less efficient than via vertex
105  auto children = particle->children(); // less efficient than via vertex
106 
108  auto px = momentum.x();
109  auto py = momentum.y();
110  auto pz = momentum.z();
111  auto et = momentum.t();
112 
114  auto vx = vertex.x();
115  auto vy = vertex.y();
116  auto vz = vertex.z();
117  auto vt = vertex.t() * 3.33564095198152022e-11; // [cm -> s]
118 
120  auto mm = parents.empty() ? -1 : parents.front()->id() - 1;
121 
123  auto ww = 1.;
124 
126  auto wt = children.empty() && st == 1;
127 
128  /* add track */
129  if (wt) {
130  primGen->AddTrack(pdg, px, py, pz, vx, vy, vz, mm, wt, et, vt, ww);
131  }
132 
133  }
136  return kTRUE;
137 }

146  if (std::abs(boost) < 1.e-6)

162 const HepMC::FourVector GeneratorHepMC::getBoostedVector(const HepMC::FourVector& vector, Double_t boost)
163 {
166  auto x = vector.x();
167  auto y = vector.y();
168  auto z = vector.z();
169  auto t = vector.t();
170  auto coshb = std::cosh(boost);
171  auto sinhb = std::sinh(boost);
172  auto xx = x;
173  auto yy = y;
174  auto zz = z * coshb - t * sinhb;
175  auto tt = t * coshb - z * sinhb;
176  return HepMC::FourVector(xx, yy, zz, tt);
177 }

201  default:

203  return kFALSE;

208 
210  return !mReader->failed();
