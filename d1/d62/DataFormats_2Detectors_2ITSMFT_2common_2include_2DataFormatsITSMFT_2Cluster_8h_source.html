176 inline void Cluster::setClusterUsage(Int_t n)
177 {
178  // set cluster usage counter
179  mNxNzN &= ~(kMaskClUse << kOffsClUse);
180  mNxNzN |= (n & kMaskClUse) << kOffsClUse;
