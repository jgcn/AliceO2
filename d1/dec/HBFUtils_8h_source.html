26 /*
27  In the MC->Raw conversion we have to make sure that
28  1) The HB and TF starts are in sync for all detectors regardless on time (bc/orbir)
29  distribution of its signal.
30  2) All HBF and TF (RAWDataHeaders with corresponding HB and TF trigger flags) are present
31  in the emulated raw data, even if some of them had no data in particular detector.
32 
33  The HBFUtils class provides tools for interaction record -> HBF conversion and sampling
34  of IRs for which the HBF RDH should be added to the raw data from the CRU.
35 
36  See testHBFUtils.cxx for the outline of generating HBF frames for simulated data.
37  */

82  /*//-------------------------------------------------------------------------------------
83  Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
84  BCs between interaction records "fromIR" and "toIR" (inclusive).
85  This method provides the IRs for RDHs to add obligatory for the MC->raw conversion,
86  in order to avoid missing HBFs (or even TFs)
87  Typical use case: assume we are converting to RAW data the digits corresponding to triggers
88  for Int.records ir[0], ir[1], ... ir[N]
89  The pseudo-code:
90 
91  HBFUtils sampler;
92  uint8_t packetCounter = 0;
93  std::vector<o2::InteractionRecord> HBIRVec;
94  auto irFrom = sampler.getFirstIR();
95  for (int i=0;i<N;i++) {
96  int nHBF = sampler.fillHBIRvector(HBIRVec, irFrom, ir[i]);
97  irFrom = ir[i]+1;
98  // nHBF-1 HBframes don't have data, we need to create empty HBFs for them
99  for (int j=0;j<nHBF-1;j++) {
100  auto rdh = sampler.createRDH<RAWDataHeader>( HBIRVec[j] );
101  // dress rdh with cruID/FEE/Link ID
102  rdh.packetCounter = packetCounter++;
103  rdh.memorySize = sizeof(rdh);
104  rdh.offsetToNext = sizeof(rdh);
105  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // open empty HBF
106  rdh.stop = 0x1;
107  rdh.pageCnt++;
108  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // close empty HBF
109  }
110  // write RDH for the HBF with data
111  auto rdh = HBIRVec.back();
112  // dress rdh with cruID/FEE/Link ID, estimate size, offset etc. and flush
113  // flush raw data payload
114  }
115  // see tstHBFUtils for more details
116  //-------------------------------------------------------------------------------------*/
117  int fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const;

138 inline o2::header::RAWDataHeaderV4 HBFUtils::createRDH<o2::header::RAWDataHeaderV4>(const o2::InteractionRecord& rec) const
139 {
140  auto tfhb = getTFandHBinTF(rec);
141  o2::header::RAWDataHeaderV4 rdh;
142  rdh.triggerBC = rec.bc;
143  rdh.triggerOrbit = rec.orbit;
144 
145  rdh.heartbeatBC = mFirstIR.bc;
146  rdh.heartbeatOrbit = rec.orbit;
147  //
148  if (rec.bc == mFirstIR.bc) { // if we are starting new HB, set the HB trigger flag
149  rdh.triggerType |= o2::trigger::HB;
150  if (tfhb.second == 0) { // if we are starting new TF, set the TF trigger flag
151  rdh.triggerType |= o2::trigger::TF;
152  }
153  }
154  return rdh;
155 }
