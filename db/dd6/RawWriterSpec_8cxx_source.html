57  auto stop = [this]() {

60  write();

67  void run(o2::framework::ProcessingContext& pc)
68  {
69  auto msg = pc.inputs().get(mInputBinding.c_str());
70  gsl::span<const ColumnData> data = of::DataRefUtils::as<const ColumnData>(msg);
71 
72  auto msgROF = pc.inputs().get(mInputROFBinding.c_str());
73  gsl::span<const ROFRecord> rofRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
74 
75  for (auto& rofRecord : rofRecords) {
76  if (rofRecord.interactionRecord.orbit != mInteractionRecord.orbit) {
77  mEncoder.newHeader(rofRecord.interactionRecord.bc, rofRecord.interactionRecord.orbit, 0);
78  mInteractionRecord = rofRecord.interactionRecord;
79  }
80  auto eventData = data.subspan(rofRecord.firstEntry, rofRecord.nEntries);
81  mEncoder.process(eventData, rofRecord.interactionRecord.bc, rofRecord.eventType);
82  }
83  write();
84  }
85 
86  private:
87  void write()
88  {
89  mFile.write(reinterpret_cast<const char*>(mEncoder.getBuffer().data()), mEncoder.getBufferSize());
90  mEncoder.clear();
91  }
