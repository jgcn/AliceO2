268 GPUd() void MEM_LG(GPUParam)::GetClusterRMS2(int iRow, float z, float sinPhi, float DzDs, float& ErrY2, float& ErrZ2) const
269 {
270  int rowType = tpcGeometry.GetROC(iRow);
271  if (rowType > 2) {
272  rowType = 2; // TODO: Add type 3
273  }
274  z = CAMath::Abs((250.f - 0.275f) - CAMath::Abs(z));
275  float s2 = sinPhi * sinPhi;
276  if (s2 > 0.95f * 0.95f) {
277  s2 = 0.95f * 0.95f;
278  }
279  float sec2 = 1.f / (1.f - s2);
280  float angleY2 = s2 * sec2; // dy/dx
281  float angleZ2 = DzDs * DzDs * sec2; // dz/dx
282 
283  ErrY2 = GetClusterRMS(0, rowType, z, angleY2);
284  ErrZ2 = GetClusterRMS(1, rowType, z, angleZ2);
285  ErrY2 *= ErrY2;
286  ErrZ2 *= ErrZ2;
